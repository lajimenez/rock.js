  3 /**
  4  * Base class for HTML components. This is a wrapper for an HTML Element that will be put over the canvas.
  5  *
  6  * @param {rock.window.Window} window
  7  *          the window where component exists
  8  * @param type
  9  *          the type of the HTML Element
 10  *
 11  * @constructor
 12  * @abstract
 13  * @extends rock.window.component.Component
 14  *
 15  * @author Luis Alberto Jim√©nez
 16  */
 17 rock.window.component.html.HTMLComponent = function (window, type) {
 18     rock.super_(this, [window]);
 19 
 20     this.width = rock.window.component.Component.AUTO_SIZE;
 21     this.height = rock.window.component.Component.AUTO_SIZE;
 22 
 23     this.HTMLElement = rock.util.DOMUtils.createHTMLElement(type);
 24     this.style = this.HTMLElement.style;
 25     this.style[rock.constants.CSS_DISPLAY] = rock.constants.CSS_DISPLAY_NONE;
 26     this.style[rock.constants.CSS_POSITION] = rock.constants.CSS_POSITION_ABSOLUTE;
 27     var canvas = this.application.getCanvas();
 28     canvas.parentNode.appendChild(this.HTMLElement);
 29 
 30     rock.util.DOMUtils.addEventListener(this.HTMLElement, rock.constants.HTML_DOM_EVENT_MOUSE_DOWN,
 31         rock.createEventHandler(this, this.dispatchOnMouseEvent, [rock.constants.ROCK_EVENT_MOUSE_DOWN]));
 32 
 33     rock.util.DOMUtils.addEventListener(this.HTMLElement, rock.constants.HTML_DOM_EVENT_MOUSE_UP,
 34         rock.createEventHandler(this, this.dispatchOnMouseEvent, [rock.constants.ROCK_EVENT_MOUSE_UP]));
 35 
 36     rock.util.DOMUtils.addEventListener(this.HTMLElement, rock.constants.HTML_DOM_EVENT_CLICK,
 37         rock.createEventHandler(this, this.dispatchOnMouseEvent, [rock.constants.ROCK_EVENT_CLICK]));
 38 
 39     rock.util.DOMUtils.addEventListener(this.HTMLElement, rock.constants.HTML_DOM_EVENT_DBLCLICK,
 40         rock.createEventHandler(this, this.dispatchOnMouseEvent, [rock.constants.ROCK_EVENT_DBLCLICK]));
 41 
 42     rock.util.DOMUtils.addEventListener(this.HTMLElement, rock.constants.HTML_DOM_EVENT_MOUSE_MOVE,
 43         rock.createEventHandler(this, this.dispatchOnMouseEvent, [rock.constants.ROCK_EVENT_MOUSE_MOVE]));
 44 
 45     rock.util.DOMUtils.addEventListener(this.HTMLElement, rock.constants.HTML_DOM_EVENT_MOUSE_OVER,
 46         rock.createEventHandler(this, this.dispatchOnMouseEvent, [rock.constants.ROCK_EVENT_MOUSE_OVER]));
 47 
 48     rock.util.DOMUtils.addEventListener(this.HTMLElement, rock.constants.HTML_DOM_EVENT_MOUSE_LEAVE,
 49         rock.createEventHandler(this, this.dispatchOnMouseEvent, [rock.constants.ROCK_EVENT_MOUSE_LEAVE]));
 50 };

 54 /**
 55  * This function will dispatch mouse events for HTML components
 56  *
 57  * @param mouseEvent
 58  *      the event thrown by the DOM
 59  * @param eventId
 60  *      id of the event to be thrown
 61  *
 62  * @private
 63  */
 64 rock.window.component.html.HTMLComponent.prototype.dispatchOnMouseEvent = function (mouseEvent, eventId) {
 65     var coords = rock.util.MouseUtils.resolveEventCoords(mouseEvent);
 66     this.dispatchEvent(
 67         rock.window.WindowSystem.createMouseEvent(
 68             eventId, coords.x, coords.y, mouseEvent
 69         )
 70     );
 71 
 72     if (eventId === rock.constants.ROCK_EVENT_MOUSE_OVER) {
 73         this.mouseOver = true;
 74     } else if (eventId === rock.constants.ROCK_EVENT_MOUSE_LEAVE) {
 75         this.mouseOver = false;
 76     }
 77 };

106 /**
107  * Set the display value of the component
108  *
109  * @param {Boolean} visible
110  */
111 rock.window.component.html.HTMLComponent.prototype.display = function(visible) {
112     if (visible) {
113         this.style[rock.constants.CSS_DISPLAY] = rock.constants.CSS_DISPLAY_BLOCK;
114 
115         // If the app has been scaled, we have to 'scale' the position
116         var scaleFactory = this.application.getScaleFactor();
117         var left = this.getX() * scaleFactory;
118         var top = this.getY() * scaleFactory;
119 
120         this.style[rock.constants.CSS_LEFT] = Math.round(left) +'px';
121         this.style[rock.constants.CSS_TOP] = Math.round(top) + 'px';
122     } else {
123         this.style[rock.constants.CSS_DISPLAY] = rock.constants.CSS_DISPLAY_NONE;
124     }
125 };

127 /**
128  * @override
129  * @see rock.graphics.engine.IDrawable#draw
130  */
131 rock.window.component.html.HTMLComponent.prototype.draw = function() {
132     this.display(true);
133     // We must assure that the element is visible, if not the computes size will be incorrect
134     this.updateComponent();
135 };

137 /**
138  * Update the size of the HTML element
139  *
140  * @param {CSSStyleDeclaration} style
141  * @param width
142  * @param height
143  * @param scaleFactor
144  * @param {Boolean} [overflow]
145  *          if true, there will be set the overflow
146  */
147 rock.window.component.html.HTMLComponent.updateHTMLStyleSize = function (style, width, height, scaleFactor, overflow) {
148     if (rock.util.JsUtils.isNullOrUndefined(width) || width < 0) {
149         style[rock.constants.CSS_WIDTH] = '';
150     } else {
151         style[rock.constants.CSS_WIDTH] = Math.round(width * scaleFactor) + 'px';
152     }
153 
154     if (rock.util.JsUtils.isNullOrUndefined(height) || height < 0) {
155         style[rock.constants.CSS_HEIGHT] = '';
156     } else {
157         style[rock.constants.CSS_HEIGHT] = Math.round(height * scaleFactor) + 'px';
158     }
159 
160     if (rock.util.JsUtils.isNullOrUndefined(overflow)) {
161         style[rock.constants.CSS_OVERFLOW] = '';
162     } else {
163         style[rock.constants.CSS_OVERFLOW] = rock.constants.CSS_OVERFLOW_HIDDEN;
164     }
165 };

167 /**
168  * Update the font of the HTML element
169  *
170  * @param {CSSStyleDeclaration} style
171  * @param {rock.graphics.Font} font
172  *            the font to use
173  * @param {rock.graphics.Color} color
174  *            the color to use
175  * @param scaleFactor
176  */
177 rock.window.component.html.HTMLComponent.updateHTMLStyleFont = function (style, font, color, scaleFactor) {
178     style[rock.constants.CSS_FONT_FAMILY] = font.getFamily();
179     style[rock.constants.CSS_FONT_SIZE] = Math.round(font.getSize() * scaleFactor) + 'px';
180     style[rock.constants.CSS_COLOR] = color.getAsHex();
181 };
