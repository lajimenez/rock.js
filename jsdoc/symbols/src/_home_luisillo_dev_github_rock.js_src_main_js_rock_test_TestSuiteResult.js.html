  3 /**
  4  * Class that contains information about execution of a test
  5  *
  6  * @constructor
  7  *
  8  * @author Luis Alberto Jim√©nez
  9  */
 10 rock.test.TestSuiteResult = function () {
 11     this.resultByTest = [];
 12 };
 13 
 14 /**
 15  * Add failed result for a test
 16  *
 17  * @param testName
 18  * @param {boolean} result
 19  */
 20 rock.test.TestSuiteResult.prototype.addResultForTest = function (testName, result) {
 21     this.resultByTest.push({name :testName, failed : result });
 22 };
 23 
 24 /**
 25  * Returns if some test has failed
 26  *
 27  * @returns {boolean} some test failed
 28  */
 29 rock.test.TestSuiteResult.prototype.someTestFailed = function () {
 30     var resultByTest = this.resultByTest;
 31     var someTestFailed = false;
 32     var i;
 33     for (i = 0; i < resultByTest.length; i++) {
 34         someTestFailed = someTestFailed || resultByTest[i].failed;
 35     }
 36     return someTestFailed;
 37 };
 38 
 39 /**
 40  * Returns execution test information
 41  *
 42  * @returns {String}
 43  */
 44 rock.test.TestSuiteResult.prototype.toString = function () {
 45     var result = '';
 46     if (this.someTestFailed()) {
 47         result += rock.resource.rockResourceManager.getString('TEST_SUITE_RESULT_KO');
 48     } else {
 49         result += rock.resource.rockResourceManager.getString('TEST_SUITE_RESULT_OK');
 50     }
 51 
 52     var i;
 53     var resultByTest = this.resultByTest;
 54     for (i = 0; i < resultByTest.length; i++) {
 55         if (resultByTest[i].failed) {
 56             result += rock.resource.rockResourceManager.getString('TEST_RESULT_KO', [resultByTest[i].name]);
 57         } else {
 58             result += rock.resource.rockResourceManager.getString('TEST_RESULT_OK', [resultByTest[i].name]);
 59         }
 60     }
 61 
 62     return result;
 63 };
