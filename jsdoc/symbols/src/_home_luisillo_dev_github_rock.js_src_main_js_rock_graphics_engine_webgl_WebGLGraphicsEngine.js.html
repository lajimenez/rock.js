  3 /**
  4  * Render engine to use with canvas and WebGL
  5  * http://www.khronos.org/registry/webgl/specs/latest/

  7  * @param {rock.graphics.engine.webgl.WebGLContext} context

 15 rock.graphics.engine.webgl.WebGLGraphicsEngine = function (context) {
 16     rock.super_(this, arguments);
 17 
 18     this.state = {
 19         saved : false
 20     };
 21 
 22     this.CIRCUMFERENCE_POINTS = 50;
 23 
 24     this.polygonRenderer = null;
 25     this.imageRenderer = null;
 26     this.floatArrayRectangle = new Float32Array(12);
 27     this.floatArrayTextureCoordiantes = new Float32Array(12);
 28     this.floatArrayCircumference = new Float32Array(3 * 2 * this.CIRCUMFERENCE_POINTS);
 29     this.circumferencePerimeter = new Float32Array(2 * this.CIRCUMFERENCE_POINTS);
 30     this.polygonRenderable = new rock.graphics.engine.renderer.PolygonRenderable();
 31     this.imageRenderable = new rock.graphics.engine.renderer.ImageRenderable();
 32 
 33     this.init();
 34     this.computePerimeterPoints();
 35 
 36     // We use a canvas2D to support operations like drawing text. We init at minimum value so
 37     // it will resize when needed
 38     var canvas2DCanvas = rock.util.DOMUtils.createCanvas(1, 1);
 39     var canvas2DContext = canvas2DCanvas.getContext(rock.constants.CANVAS_CONTEXT_2D);
 40     var canvas2DGraphicsEngineContext = new rock.graphics.engine.canvas2d.Canvas2DContext(this, canvas2DContext);
 41     this.canvasGraphicsEngine = new rock.graphics.engine.canvas2d.Canvas2DGraphicsEngine(canvas2DGraphicsEngineContext);
 42 
 43     this.transparent = rock.graphics.Color.TRANSPARENT;
 44 };

 58 /**
 59  * We need some WebGL conditions to do engine work properly
 60  *
 61  * @param {Boolean} [premultipliedAlpha]
 62  *      if the color used for drawing has premultiplied alpha
 63  *
 64  * @private
 65  */
 66 rock.graphics.engine.webgl.WebGLGraphicsEngine.prototype.prepareWebGLEngine = function (premultipliedAlpha) {
 67     var gl = this.context.getHTMLContext();
 68     gl.enable(gl.CULL_FACE);
 69     gl.frontFace(gl.CCW);
 70     gl.cullFace(gl.BACK);
 71     // Color mask set to write all colors
 72     gl.colorMask(true, true, true, true);
 73     // Last draw will be visible
 74     gl.disable(gl.DEPTH_TEST);
 75     // We want blending
 76     gl.enable(gl.BLEND);
 77     gl.blendEquation(gl.FUNC_ADD);
 78     if (premultipliedAlpha) {
 79         gl.blendFuncSeparate(gl.ONE, gl.ONE_MINUS_SRC_ALPHA, gl.ONE, gl.ONE);
 80     } else {
 81         gl.blendFuncSeparate(gl.SRC_ALPHA, gl.ONE_MINUS_SRC_ALPHA, gl.ONE, gl.ONE);
 82     }
 83 };

 85 /**
 86  * Save the current state of the WebGL.
 87  * You could call this function before doing a graphics API call and so you will be able to restore the WebGL state
 88  */
 89 rock.graphics.engine.webgl.WebGLGraphicsEngine.prototype.saveState = function () {
 90     var gl = this.context.getHTMLContext();
 91     var state = this.state;
 92     state.colorMask = gl.getParameter(gl.COLOR_WRITEMASK);
 93     state.cullFaceEnabled = gl.getParameter(gl.CULL_FACE);
 94     state.frontFace = gl.getParameter(gl.FRONT_FACE);
 95     state.cullFace = gl.getParameter(gl.CULL_FACE_MODE);
 96     state.depthTestEnabled = gl.getParameter(gl.DEPTH_TEST);
 97     state.blendEnabled = gl.getParameter(gl.BLEND);
 98     state.blendEquationRGB = gl.getParameter(gl.BLEND_EQUATION_RGB);
 99     state.blendEquationAlpha = gl.getParameter(gl.BLEND_EQUATION_ALPHA);
100     state.blendFuncSrcRGB = gl.getParameter(gl.BLEND_SRC_RGB);
101     state.blendFuncSrcAlpha = gl.getParameter(gl.BLEND_SRC_ALPHA);
102     state.blendFuncDstRGB = gl.getParameter(gl.BLEND_DST_RGB);
103     state.blendFuncDstAlpha = gl.getParameter(gl.BLEND_DST_ALPHA);
104     state.saved = true;
105 };

107 /**
108  * Restore the WebGL state (a state must have been saved before call this method)
109  */
110 rock.graphics.engine.webgl.WebGLGraphicsEngine.prototype.restoreState = function () {
111     var state = this.state;
112     if (!state.saved) {
113         return;
114     }
115 
116     var gl = this.context.getHTMLContext();
117 
118     var colorMask = state.colorMask;
119     gl.colorMask(colorMask[0], colorMask[1], colorMask[2], colorMask[3]);
120 
121     if (state.cullFaceEnabled) {
122         gl.enable(gl.CULL_FACE);
123     } else {
124         gl.disable(gl.CULL_FACE);
125     }
126 
127     gl.frontFace(state.frontFace);
128     gl.cullFace(state.cullFace);
129 
130     if (state.depthTestEnabled) {
131         gl.enable(gl.DEPTH_TEST);
132     } else {
133         gl.disable(gl.DEPTH_TEST);
134     }
135 
136     if (state.blendEnabled) {
137         gl.enable(gl.BLEND);
138     } else {
139         gl.disable(gl.BLEND);
140     }
141 
142     gl.blendEquationSeparate(state.blendEquationRGB, state.blendEquationAlpha);
143     gl.blendFuncSeparate(state.blendFuncSrcRGB, state.blendFuncDstRGB,
144         state.blendFuncSrcAlpha, state.blendFuncDstAlpha);
145 };

158 rock.graphics.engine.webgl.WebGLGraphicsEngine.prototype.computePerimeterPoints = function () {
159     var circumferencePerimeter = this.circumferencePerimeter;
160     var i, rad;
161     var TWO_PI = (2 * Math.PI);
162     var circumferencePoints = this.CIRCUMFERENCE_POINTS;
163     var difference = TWO_PI / circumferencePoints;
164 
165     for (i = 0; i < circumferencePoints; i++) {
166         rad = i * difference;
167         circumferencePerimeter[2 * i] = Math.cos(rad);
168         circumferencePerimeter[2 * i + 1] = Math.sin(rad);
169     }
170 };

172 rock.graphics.engine.webgl.WebGLGraphicsEngine.prototype.computeCircumferenceVertices = function (x, y, size) {
173     var radius = size / 2.;
174     var circumferencePerimeter = this.circumferencePerimeter;
175     var circumferencePoints = this.CIRCUMFERENCE_POINTS;
176     var vertices = this.floatArrayCircumference;
177     var i, pos = 0, currentPointX, currentPointY, nextPointX, nextPointY;
178 
179     for (i = 0; i < circumferencePoints - 1; i++) {
180         pos = 2 * i;
181         currentPointX = circumferencePerimeter[pos];
182         currentPointY = circumferencePerimeter[pos + 1];
183         nextPointX = circumferencePerimeter[pos + 2];
184         nextPointY = circumferencePerimeter[pos + 3];
185 
186         pos = 3 * 2 * i;
187         vertices[pos] = x;
188         vertices[pos + 1] = y;
189 
190         vertices[pos + 2] = x + nextPointX * radius;
191         vertices[pos + 3] = y + nextPointY * radius;
192 
193         vertices[pos + 4] = x + currentPointX * radius;
194         vertices[pos + 5] = y + currentPointY * radius;
195     }
196 
197     pos = 3 * 2 * (circumferencePoints - 1);
198     vertices[pos] = x;
199     vertices[pos + 1] = y;
200 
201     vertices[pos + 2] = x + circumferencePerimeter[0] * radius;
202     vertices[pos + 3] = y + circumferencePerimeter[1] * radius;
203 
204     vertices[pos + 4] = x + circumferencePerimeter[2 * circumferencePoints - 2] * radius;
205     vertices[pos + 5] = y + circumferencePerimeter[2 * circumferencePoints - 1] * radius;
206 };

217 /**
218  * @see rock.graphics.engine.IGraphicsEngine#drawLine
219  * @override
220  */
221 rock.graphics.engine.webgl.WebGLGraphicsEngine.prototype.drawLine = function (fromX, fromY, toX, toY, width, color) {
222     var halfWidth = width / 2.;
223     var orientation = Math.atan2(toY - fromY, toX - fromX);
224     var halfPI = Math.PI / 2.;
225 
226     var vertices = this.floatArrayRectangle;
227     vertices[0] = fromX + (halfWidth * Math.cos(orientation - halfPI));
228     vertices[1] = fromY + (halfWidth * Math.sin(orientation - halfPI));
229 
230     vertices[2] = fromX + (halfWidth * Math.cos(orientation + halfPI));
231     vertices[3] = fromY + (halfWidth * Math.sin(orientation + halfPI));
232 
233     vertices[4] = toX + (halfWidth * Math.cos(orientation + halfPI));
234     vertices[5] = toY + (halfWidth * Math.sin(orientation + halfPI));
235 
236     vertices[6] = fromX + (halfWidth * Math.cos(orientation - halfPI));
237     vertices[7] = fromY + (halfWidth * Math.sin(orientation - halfPI));
238 
239     vertices[8] = toX + (halfWidth * Math.cos(orientation + halfPI));
240     vertices[9] = toY + (halfWidth * Math.sin(orientation + halfPI));
241 
242     vertices[10] = toX + (halfWidth * Math.cos(orientation - halfPI));
243     vertices[11] = toY + (halfWidth * Math.sin(orientation - halfPI));
244 
245     this.adjustVertices(vertices, width);
246 
247     this.renderPolygon(vertices, 6, color);
248 };

250 rock.graphics.engine.webgl.WebGLGraphicsEngine.prototype.adjustVertices = function (vertices, width) {
251     var i, value;
252     for (i = 0; i < vertices.length; i++) {
253         value = vertices[i];
254         vertices[i] = this.adjustPixelPosition(value, width);
255     }
256 };

328 rock.graphics.engine.webgl.WebGLGraphicsEngine.prototype.drawStrokeRectangle = function
329     (x, y, width, height, color, strokeWidth) {
330     var middleStrokeWidth = strokeWidth / 2;
331     var xmin = x;
332     var xmax = x + width;
333     var ymin = y;
334     var ymax = y + height;
335     this.drawLine(xmin - middleStrokeWidth, ymin, xmax + middleStrokeWidth, ymin, strokeWidth, color);
336     this.drawLine(xmin, ymin, xmin, ymax, strokeWidth, color);
337     this.drawLine(xmin - middleStrokeWidth, ymax, xmax + middleStrokeWidth, ymax, strokeWidth, color);
338     this.drawLine(xmax, ymin, xmax, ymax, strokeWidth, color);
339 };

359

361     this.drawFilledRectangle(x + radius, y, width - size, height, color);

392  * @see rock.graphics.engine.IGraphicsEngine#drawImage
393  * @override
394  */
395 rock.graphics.engine.webgl.WebGLGraphicsEngine.prototype.drawImage = function (image, x, y, width, height) {
396     var drawingWidth, drawingHeight;
397 
398     if (!rock.util.JsUtils.isNullOrUndefined(width)) {
399         drawingWidth = width;
400     } else {
401         drawingWidth = image.getWidth();
402     }
403 
404     if (!rock.util.JsUtils.isNullOrUndefined(height)) {
405         drawingHeight = height;
406     } else {
407         drawingHeight = image.getHeight();
408     }
409 
410     var vertices = this.floatArrayRectangle;
411     vertices[0] = x;
412     vertices[1] = y;
413 
414     vertices[2] = x;
415     vertices[3] = y + drawingHeight;
416 
417     vertices[4] = x + drawingWidth;
418     vertices[5] = y + drawingHeight;
419 
420     vertices[6] = x;
421     vertices[7] = y;
422 
423     vertices[8] = x + drawingWidth;
424     vertices[9] = y + drawingHeight;
425 
426     vertices[10] = x + drawingWidth;
427     vertices[11] = y;
428 
429     var textureVertices = this.floatArrayTextureCoordiantes;
430     textureVertices[0] = 0;
431     textureVertices[1] = 0;
432 
433     textureVertices[2] = 0;
434     textureVertices[3] = 1;
435 
436     textureVertices[4] = 1;
437     textureVertices[5] = 1;
438 
439     textureVertices[6] = 0;
440     textureVertices[7] = 0;
441 
442     textureVertices[8] = 1;
443     textureVertices[9] = 1;
444 
445     textureVertices[10] = 1;
446     textureVertices[11] = 0;
447 
448     this.renderImage(image, vertices, 6, textureVertices);
449 };

451 /**
452  * @see rock.graphics.engine.IGraphicsEngine#drawSubImage
453  * @override
454  */
455 rock.graphics.engine.webgl.WebGLGraphicsEngine.prototype.drawSubImage = function (image, x, y, subX, subY, subWidth, subHeight) {
456     var width = image.getWidth();
457     var height = image.getHeight();
458 
459     var imageRectangle = new rock.geometry.Rectangle(x, y, x + width, y + height);
460     var drawingRectangle = new rock.geometry.Rectangle(x + subX, y + subY,
461         x + subX + subWidth, y + subY + subHeight);
462 
463     var intersection = imageRectangle.getIntersectionWithRectangle(drawingRectangle);
464     if (rock.util.JsUtils.isNullOrUndefined(intersection)) {
465         return;
466     }
467 
468     var intersectionWidth = intersection.getWidth();
469     var intersectionHeight = intersection.getHeight();
470 
471     var textMinX = subX / width;
472     var textMaxX = (subX + intersectionWidth) / width;
473     var textMinY = subY / height;
474     var textMaxY = (subY + intersectionHeight) / height;
475 
476     var vertices = this.floatArrayRectangle;
477     vertices[0] = x;
478     vertices[1] = y;
479 
480     vertices[2] = x;
481     vertices[3] = y + intersectionHeight;
482 
483     vertices[4] = x + intersectionWidth;
484     vertices[5] = y + intersectionHeight;
485 
486     vertices[6] = x;
487     vertices[7] = y;
488 
489     vertices[8] = x + intersectionWidth;
490     vertices[9] = y + intersectionHeight;
491 
492     vertices[10] = x + intersectionWidth;
493     vertices[11] = y;
494 
495     var textureCoordinates = this.floatArrayTextureCoordiantes;
496     textureCoordinates[0] = textMinX;
497     textureCoordinates[1] = textMinY;
498 
499     textureCoordinates[2] = textMinX;
500     textureCoordinates[3] = textMaxY;
501 
502     textureCoordinates[4] = textMaxX;
503     textureCoordinates[5] = textMaxY;
504 
505     textureCoordinates[6] = textMinX;
506     textureCoordinates[7] = textMinY;
507 
508     textureCoordinates[8] = textMaxX;
509     textureCoordinates[9] = textMaxY;
510 
511     textureCoordinates[10] = textMaxX;
512     textureCoordinates[11] = textMinY;
513 
514     this.renderImage(image, vertices, 6, textureCoordinates);
515 };

547 /**
548  * @see rock.graphics.engine.IGraphicsEngine#drawText
549  * @override
550  */
551 rock.graphics.engine.webgl.WebGLGraphicsEngine.prototype.drawText = function
552         (text, x, y, font, color, clippingWidth, clippingHeight) {
553 
554     var textWidth = rock.util.TextUtils.measureTextWidth(text, font);
555     var textHeight = rock.util.TextUtils.measureTextHeight(text, font);
556 
557     if (textWidth == 0 || textHeight == 0) {
558         return;
559     }
560 
561     // we compute maximum possible width & height
562     var width = this.width - x;
563     var height = this.height - y;
564 
565     if (textWidth > width) {
566         textWidth = width;
567     }
568 
569     if (textHeight > height) {
570         textHeight = height;
571     }
572 
573     var canvasGraphicsEngine = this.canvasGraphicsEngine;
574     var canvas2D = canvasGraphicsEngine.getContext().getHTMLContext();
575     canvas2D.canvas.width = textWidth;
576     canvas2D.canvas.height = textHeight;
577     canvasGraphicsEngine.computeSize();
578 
579     canvasGraphicsEngine.clear(this.transparent);
580     canvasGraphicsEngine.drawText(text, 0, 0, font, color, clippingWidth, clippingHeight);
581 
582     var imageData = canvas2D.getImageData(0, 0, textWidth, textHeight);
583     var image = new rock.graphics.Image(imageData);
584 
585     this.drawImage(image, x, y);
586 };
