  3 /**
  4  * WebGL utils class
  5  *
  6  * @constructor
  7  * @author Luis Alberto Jim√©nez

  9 rock.util.WebGlUtils = {
 10     /**
 11      * Create and compile a shader
 12      *
 13      * @param gl
 14      *            the WebGL context
 15      * @param type
 16      *            the type of the shader {gl.VERTEX_SHADER, gl.FRAGMENT_SHADER}
 17      * @param src
 18      *            the shader source
 19      * @returns the shader
 20      */
 21     createAndCompileShader: function (gl, type, src) {
 22         var shader = gl.createShader(type);
 23         gl.shaderSource(shader, src);
 24         gl.compileShader(shader);
 25         if (!gl.getShaderParameter(shader, gl.COMPILE_STATUS) && !gl.isContextLost()) {
 26             // var shaderInfoLog = gl.getShaderInfoLog(shader);
 27             throw new rock.error.RockError(rock.resource.rockResourceManager.getString('SHADER_NOT_COMPILED'));
 28         }
 29         return shader;
 30     },
 31 
 32     /**
 33      * Creates a new program
 34      *
 35      * @param gl
 36      *            the webgl context
 37      * @param vertexShader
 38      *            the vertex shader to use in the program
 39      * @param fragmentShader
 40      *            the fragment shader to use in the program
 41      * @returns the program
 42      */
 43     createProgram: function (gl, vertexShader, fragmentShader) {
 44         var program = gl.createProgram();
 45         gl.attachShader(program, vertexShader);
 46         gl.attachShader(program, fragmentShader);
 47         this.linkProgram(gl, program);
 48         return program;
 49     },
 50 
 51     /**
 52      * Links a program
 53      *
 54      * @param gl
 55      *            the webgl context
 56      * @param program
 57      *            the program to link
 58      */
 59     linkProgram: function (gl, program) {
 60         gl.linkProgram(program);
 61         if (!gl.getProgramParameter(program, gl.LINK_STATUS) && !gl.isContextLost()) {
 62             // gl.getProgramInfoLog(program);
 63             throw new rock.error.RockError(rock.resource.rockResourceManager.getString('PROGRAM_NOT_LINKED'));
 64         }
 65     },
 66 
 67     /**
 68      * Populate an attribute in a GLSL program. The buffer used for populating
 69      * attribute will be bound.
 70      *
 71      * @param gl
 72      *            the WebGL context
 73      * @param program
 74      *            the program that have the attribute
 75      * @param attribute
 76      *            attribute name
 77      * @param vertices
 78      *            array of vertices
 79      * @param dimensions
 80      *            dimensions of the vertices
 81      * @returns {WebGLBuffer} the buffer used for populate attribute
 82      */
 83     populateArrayBufferAttribute: function (gl, program, attribute, vertices, dimensions) {
 84         var buffer = gl.createBuffer();
 85         gl.bindBuffer(gl.ARRAY_BUFFER, buffer);
 86         gl.bufferData(gl.ARRAY_BUFFER, new Float32Array(vertices), gl.STATIC_DRAW);
 87 
 88         var attribLocation = gl.getAttribLocation(program, attribute);
 89         gl.enableVertexAttribArray(attribLocation);
 90         gl.vertexAttribPointer(attribLocation, dimensions, gl.FLOAT, gl.FALSE, 0, 0);
 91 
 92         return buffer;
 93     },
 94 
 95     /**
 96      * Populate a indexes buffer used to draw elements. The buffer created will
 97      * be bound
 98      *
 99      * @param gl
100      *            the WebGL context
101      * @param indexes
102      *            array with indexes
103      *
104      * @returns {WebGLBuffer} the buffer created to draw
105      */
106     populateElementArrayBuffer: function (gl, indexes) {
107         var buffer = gl.createBuffer();
108         gl.bindBuffer(gl.ELEMENT_ARRAY_BUFFER, buffer);
109         gl.bufferData(gl.ELEMENT_ARRAY_BUFFER, new Uint16Array(indexes), gl.STATIC_DRAW);
110         return buffer;
111     },
112 
113     /**
114      * Create and populate a texture
115      *
116      * @param gl
117      *            the WebGL context
118      * @param {rock.graphics.Image} image
119      *            the image used for the texture
120      *
121      * @returns {WebGLTexture} the texture created
122      */
123     populateTexture: function (gl, image) {
124         // Create a texture.
125         var texture = gl.createTexture();
126         gl.bindTexture(gl.TEXTURE_2D, texture);
127         // Configure how to draw texture
128         gl.texParameteri(gl.TEXTURE_2D, gl.TEXTURE_WRAP_S, gl.CLAMP_TO_EDGE);
129         gl.texParameteri(gl.TEXTURE_2D, gl.TEXTURE_WRAP_T, gl.CLAMP_TO_EDGE);
130         gl.texParameteri(gl.TEXTURE_2D, gl.TEXTURE_MIN_FILTER, gl.NEAREST);
131         gl.texParameteri(gl.TEXTURE_2D, gl.TEXTURE_MAG_FILTER, gl.NEAREST);
132 
133         // Upload the image into the texture.
134         gl.texImage2D(gl.TEXTURE_2D, 0, gl.RGBA, gl.RGBA, gl.UNSIGNED_BYTE,
135             image.getHTMLImage());
136         return texture;
137     },
138 
139     /**
140      * Disable all vertex attribute in a program
141      *
142      * @param gl
143      *            the WebGL context
144      * @param program
145      *            the program to disable vertext attribute
146      */
147     disableAllVertexAttribArray : function (gl, program) {
148         var currentAttributes = gl.getProgramParameter(program, gl.ACTIVE_ATTRIBUTES);
149         var i;
150         for (i = 0; i < currentAttributes; i++) {
151             gl.disableVertexAttribArray(i);
152         }
153     }
154 };
