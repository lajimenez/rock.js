 12     /**
 13      * Get the coordinates of an event. The position is relative to the current
 14      * element.
 15      *
 16      * @param event
 17      *            the mouse event
 18      */
 19     resolveEventCoords: function (event) {
 20         var boundingClientRect = event.currentTarget.getBoundingClientRect();
 21 
 22         return {
 23             x: event.clientX - boundingClientRect.left,
 24             y: event.clientY - boundingClientRect.top
 25         };
 26     },
 27 
 28     resolveEventCoords_CH: function (event) {
 29         return {
 30             x: event.offsetX,
 31             y: event.offsetY
 32         };
 33     },
 34

 36 };
