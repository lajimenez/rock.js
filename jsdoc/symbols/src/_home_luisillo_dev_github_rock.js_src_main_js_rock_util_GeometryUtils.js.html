  9 rock.util.GeometryUtils = {
 10     /**
 11      * Convert an angle from degrees to radians
 12      *
 13      * @param degrees
 14      *
 15      * @returns {Number} angle in radians
 16      */
 17     degToRad: function (degrees) {
 18         return (degrees * 2 * Math.PI) / 360;
 19     },
 20 
 21     /**
 22      * Convert an angle from radians to degrees
 23      *
 24      * @param radians
 25      *
 26      * @returns {Number} angle in degrees
 27      */
 28     radToDeg: function (radians) {
 29         return (radians * 360) / (2 * Math.PI);
 30     },
 31 
 32     /**
 33      * Cross product between 2 vectors
 34      *
 35      * @param {rock.geometry.Vector3} vector1
 36      *            vector 1
 37      * @param {rock.geometry.Vector3} vector2
 38      *            vector 2
 39      * @param {rock.geometry.Vector3} [result_ma]
 40      *            to avoid memory allocation, you can pass a vector that will be updated instead of creating a new
 41      *            as a result
 42      * @returns {rock.geometry.Vector3} the result vector
 43      */
 44     crossProduct: function (vector1, vector2, result_ma) {
 45         var result = result_ma;
 46         if (rock.util.JsUtils.isNullOrUndefined(result)) {
 47             result = new rock.geometry.Vector3(0, 0, 0);
 48         }
 49 
 50         result.setX((vector1.getY() * vector2.getZ()) - (vector1.getZ() * vector2.getY()));
 51 
 52         result.setY((vector1.getZ() * vector2.getX()) - (vector1.getX() * vector2.getZ()));
 53 
 54         result.setZ((vector1.getX() * vector2.getY()) - (vector1.getY() * vector2.getX()));
 55 
 56         return result;
 57     },
 58 
 59     /**
 60      * Return the point that makes share center point between 2 elements
 61      * (Element has the 'starting' point, width and height
 62      *
 63      * @param {rock.geometry.Point2} point
 64      *      point where 'base' element start
 65      * @param width
 66      *      width for 'base' element
 67      * @param height
 68      *      height for 'base' element
 69      * @param widthElemToCenter
 70      *      width for element we want to share center with base element
 71      * @param heightElemToCenter
 72      *      height for element we want to share center with base element
 73      * @returns {rock.geometry.Point2} start point for element we want to share center
 74      */
 75     getPointMakesSameCenter: function (point, width, height, widthElemToCenter, heightElemToCenter) {
 76         var x = point.getX() + ((width - widthElemToCenter) / 2);
 77         var y = point.getY() + ((height - heightElemToCenter) / 2);
 78         return new rock.geometry.Point2(Math.round(x), Math.round(y));
 79     },
 80 
 81     /**
 82      * Normalize and angle in degrees (0 to 360)
 83      *
 84      * @param angle
 85      *      the angle to normalize
 86      * @return {Number} the angle normalized
 87      */
 88     normalizeAngle : function(angle) {
 89         var normalizedAngle = angle;
 90         while (normalizedAngle < 0) {
 91             normalizedAngle += 360;
 92         }
 93         normalizedAngle = normalizedAngle % 360;
 94         return normalizedAngle;
 95     }

 97 };
