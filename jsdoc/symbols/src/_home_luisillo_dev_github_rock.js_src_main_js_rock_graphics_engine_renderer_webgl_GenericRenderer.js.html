  3 /**
  4  * Generic WebGL renderer. It helps to cache locations, buffers and textures and has a generic render function.
  5  *
  6  * @param {rock.graphics.engine.webgl.WebGLContext} glContext
  7  *         the WebGL context
  8  * @param {String} vertexShaderSrc
  9  *         source for the wertex shader
 10  * @param {String} fragmentShaderSrc
 11  *         source for the fragment shader
 12  * @param {Array} attributes
 13  *          list of attributes in program
 14  * @param {Array} uniforms
 15  *          list of uniform in program
 16  * @param {Array} textures
 17  *          list of textures in program
 18  *
 19  * @constructor
 20  * @extends {rock.graphics.engine.renderer.webgl.Renderer}
 21  *
 22  * @author Luis Alberto Jim√©nez
 23  */
 24 rock.graphics.engine.renderer.webgl.GenericRenderer = function (glContext, vertexShaderSrc, fragmentShaderSrc,
 25                                                                          attributes, uniforms, textures) {
 26     rock.super_(this, arguments);
 27     var gl = glContext.getHTMLContext();
 28     this.glDrawMode = gl.TRIANGLES;
 29 };
 30 
 31 rock.extends_(rock.graphics.engine.renderer.webgl.GenericRenderer,
 32     rock.graphics.engine.renderer.webgl.Renderer);
 33 
 34 /**
 35  * @see rock.graphics.engine.renderer.IRenderer#render
 36  * @override
 37  */
 38 rock.graphics.engine.renderer.webgl.GenericRenderer.prototype.render = function (renderable) {
 39     var program = this.program;
 40     program.useProgram();
 41 
 42     var gl = this.glContext.getHTMLContext();
 43 
 44     program.enableAttributes();
 45 
 46     this.populateAttributes(renderable);
 47     this.populateTextures(renderable);
 48     this.setUniforms(renderable);
 49 
 50     gl.drawArrays(this.glDrawMode, 0, renderable.getCount());
 51 
 52     gl.bindBuffer(gl.ARRAY_BUFFER, null);
 53     gl.bindTexture(gl.TEXTURE_2D, null);
 54 
 55     program.disableAttributes();
 56 };
