  3 /**
  4  * Represents a rectangle
  5  *
  6  * @param xMin
  7  * @param yMin
  8  * @param xMax
  9  * @param yMax
 10  *
 11  * @constructor
 12  *

 15 rock.geometry.Rectangle = function (xMin, yMin, xMax, yMax) {
 16     this.xMin = xMin;
 17     this.yMin = yMin;
 18     this.xMax = xMax;
 19     this.yMax = yMax;
 20 };

 22 /**
 23  * Return if the point intersects with the rectangle
 24  *
 25  * @param {rock.geometry.Point2} point
 26  *
 27  * @returns {boolean}
 28  */
 29 rock.geometry.Rectangle.prototype.intersectWithPoint = function (point) {
 30     var pointX = point.getX();
 31     var pointY = point.getY();
 32 
 33     return this.xMin <= pointX && pointX <= this.xMax && this.yMin <= pointY && pointY <= this.yMax;
 34 };

 36 /**
 37  * Return the intersection between 2 rectangles
 38  *
 39  * @param {rock.geometry.Rectangle} rectangle
 40  *
 41  * @returns {null |rock.geometry.Rectangle }
 42  */
 43 rock.geometry.Rectangle.prototype.getIntersectionWithRectangle = function (rectangle) {
 44     var result = null;
 45     var intersectXMin = rectangle.getXMin();
 46     var intersectYMin = rectangle.getYMin();
 47     var intersectXMax = rectangle.getXMax();
 48     var intersectYMax = rectangle.getYMax();
 49     var xMin = this.xMin;
 50     var yMin = this.yMin;
 51     var xMax = this.xMax;
 52     var yMax = this.yMax;
 53 
 54     if (xMax < intersectXMin || intersectXMax < xMin
 55         || yMax < intersectYMin || intersectYMax < yMin) {
 56         return result;
 57     }
 58 
 59     var resultXMin = xMin;
 60     if (intersectXMin > resultXMin) {
 61         resultXMin = intersectXMin;
 62     }
 63 
 64     var resultYMin = yMin;
 65     if (intersectYMin > resultYMin) {
 66         resultYMin = intersectYMin;
 67     }
 68 
 69     var resultXMax = xMax;
 70     if (intersectXMax < resultXMax) {
 71         resultXMax = intersectXMax;
 72     }
 73 
 74     var resultYMax = yMax;
 75     if (intersectYMax < resultYMax) {
 76         resultYMax = intersectYMax;
 77     }
 78 
 79     // Intersection might be a line or point, but this function will always return a Rectangle.
 80     return new rock.geometry.Rectangle(resultXMin, resultYMin, resultXMax, resultYMax);
 81 };

 83 /**
 84  * Get the width of the rectangle
 85  *
 86  * @returns {number}
 87  */
 88 rock.geometry.Rectangle.prototype.getWidth = function () {
 89     return Math.abs(this.xMax - this.xMin);
 90 };
 91 
 92 /**
 93  * Get the height of the rectangle
 94  *
 95  * @returns {number}
 96  */
 97 rock.geometry.Rectangle.prototype.getHeight = function () {
 98     return Math.abs(this.yMax - this.yMin);
 99 };
