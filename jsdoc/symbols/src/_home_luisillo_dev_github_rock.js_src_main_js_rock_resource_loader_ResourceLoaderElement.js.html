  3 /**
  4  * Information for resource loading
  5  *
  6  * @param id
  7  *            the id
  8  * @param type
  9  *            resource type
 10  * @param {rock.network.HTTPRequestParams} HTTPParams
 11  *            resource HTTP Params to use.
 12  *            ATTENTION: There are some resources that WILL NOT USE all HTTP params and only will use url.
 13  *            (for example images)
 14  * @param [weight]
 15  *            specific weight for this element, can be undefined
 16  */
 17 rock.resource.loader.ResourceLoaderElement = function (id, type, HTTPParams, weight) {
 18     this.id = id;
 19     this.type = type;
 20     this.HTTPParams = HTTPParams;
 21     this.weight = weight;
 22 };
 23 
 24 /**
