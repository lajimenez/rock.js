  3 /**
  4  * Represents a WebGL shader
  5  *
  6  * @param {rock.graphics.engine.webgl.WebGLContext} glContext
  7  *         the WebGL context
  8  * @param {String} src
  9  *         the source of the shader
 10  * @param {String} type
 11  *         the type of the shader. Must be [VERTEX_SHADER | FRAGMENT_SHADER]
 12  *
 13  * @constructor
 14  *
 15  * @author Luis Alberto Jim√©nez
 16  */
 17 rock.graphics.engine.renderer.webgl.program.Shader = function (glContext, src, type) {
 18     this.glContext = glContext;
 19     this.src = src;
 20     this.type = type;
 21     this.compiled = false;
 22     this.glShader = null;
 23 };
 24 
 25 /**
 26  * Vertex shader type constant
 27  */
 28 rock.graphics.engine.renderer.webgl.program.Shader.VERTEX_SHADER = 'VERTEX_SHADER';
 29 
 30 /**
 31  * Fragment shader type constant
 32  */
 33 rock.graphics.engine.renderer.webgl.program.Shader.FRAGMENT_SHADER = 'FRAGMENT_SHADER';
 34 
 35 /**
 36  * Compile the shader
 37  */
 38 rock.graphics.engine.renderer.webgl.program.Shader.prototype.compile = function () {
 39     var gl = this.glContext.getHTMLContext();
 40 
 41     var glType = gl.VERTEX_SHADER;
 42     if (this.type === rock.graphics.engine.renderer.webgl.program.Shader.FRAGMENT_SHADER) {
 43         glType = gl.FRAGMENT_SHADER;
 44     }
 45 
 46     this.glShader = rock.util.WebGlUtils.createAndCompileShader(gl, glType, this.src);
 47     this.compiled = true;
 48 };
 49

 64 /**
