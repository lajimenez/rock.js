  3 /**
  4  * Abstract class for components that has some text to show
  5  *
  6  * (This class is almost a copy of the class {rock.window.component.TextComponent}...
  7  * I should do some refactor to avoid it...)
  8  *
  9  * @param {rock.window.Window} window
 10  *            the window where component exists
 11  * @param {String} text
 12  *            the text to show in label
 13  * @param {Boolean} [localize]
 14  *            if true, the text will be localized (so text should be the resource id of the localized string)
 15  * @param {rock.graphics.Font} [font]
 16  *            the font to use
 17  * @param {rock.graphics.Color} [color]
 18  *            the color to use
 19  * @param type
 20  *          the type of the HTML Element
 21  *
 22  *
 23  * @constructor
 24  * @abstract
 25  * @extends rock.window.component.html.HTMLComponent
 26  *
 27  * @author Luis Alberto Jim√©nez
 28  */
 29 rock.window.component.html.HTMLTextComponent = function (window, text, localize, font, color, type) {
 30     rock.super_(this, [window, type]);
 31 
 32     this.text = text;
 33     if (rock.util.JsUtils.isNullOrUndefined(this.text)) {
 34         this.text = '';
 35     }
 36     this.localize = localize;
 37     if (rock.util.JsUtils.isNullOrUndefined(this.localize)) {
 38         this.localize = false;
 39     }
 40     this.font = font;
 41     if (rock.util.JsUtils.isNullOrUndefined(this.font)) {
 42         this.font = this.application.getStyleManager().getFont(rock.window.style.StyleManager.COMPONENT_FONT);
 43     }
 44     this.color = color;
 45     if (rock.util.JsUtils.isNullOrUndefined(this.color)) {
 46         this.color = this.application.getStyleManager().getColor(rock.window.style.StyleManager.COMPONENT_FONT_COLOR);
 47     }
 48 
 49     // This is the real text that will be shown
 50     this.localizedText = null;
 51     this.localizeText();
 52 };
 53 
 54 rock.extends_(rock.window.component.html.HTMLTextComponent, rock.window.component.html.HTMLComponent);
 55 
 56 /**
 57  * Get the text to use
 58  */
 59 rock.window.component.html.HTMLTextComponent.prototype.localizeText = function () {
 60     var localizedText = this.text;
 61     if (this.localize) {
 62         localizedText = this.application.getResourceManager().getString(this.text);
 63     }
 64     this.localizedText = localizedText;
 65 };
