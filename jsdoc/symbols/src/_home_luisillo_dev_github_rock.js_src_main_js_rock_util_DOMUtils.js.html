  3 /**
  4  * DOM utils class
  5  *
  6  * @constructor
  7  * @author Luis Alberto Jim√©nez
  8  */
  9 rock.util.DOMUtils = {
 10 
 11     /**
 12      * Get a HTML Element by id
 13      *
 14      * @param id
 15      *            the id of the HTML Element
 16      */
 17     getElementById: function (id) {
 18         return document.getElementById(id);
 19     },
 20 
 21     /**
 22      * Add an event listener to an HTML Element
 23      *
 24      * @param element
 25      *            the HTML element
 26      * @param eventType
 27      *            the event to add listener
 28      * @param handler
 29      *            the handler for the event
 30      * @param {Boolean} [useCapture]
 31      *            use capture
 32      */
 33     addEventListener: function (element, eventType, handler, useCapture) {
 34         var capture = false;
 35         if (!rock.util.JsUtils.isNullOrUndefined(useCapture)) {
 36             capture = useCapture;
 37         }
 38         element.addEventListener(eventType, handler, capture);
 39     },
 40 
 41     /**
 42      * Remove an event listener from an HTML Element
 43      *
 44      * @param element
 45      *            the HTML element
 46      * @param eventType
 47      *            the event to add listener
 48      * @param handler
 49      *            the handler for the event
 50      * @param {Boolean} [useCapture]
 51      *            use capture
 52      */
 53     removeEventListener: function (element, eventType, handler, useCapture) {
 54         var capture = false;
 55         if (!rock.util.JsUtils.isNullOrUndefined(useCapture)) {
 56             capture = useCapture;
 57         }
 58         element.removeEventListener(eventType, handler, capture);
 59     },
 60 
 61     /**
 62      * Create a HTML Element
 63      *
 64      * @param type
 65      *            the type of the HTML Element
 66      */
 67     createHTMLElement: function (type) {
 68         var HTMLElement = document.createElement(type);
 69         return HTMLElement;
 70     },
 71 
 72     /**
 73      * Create a canvas element
 74      *
 75      * @param width
 76      *            the width of the canvas
 77      * @param height
 78      *            the height of the canvas
 79      * @param {number} [tabIndex]
 80      *            the tab index. It's a trick to be able to set focus on the element...
 81      */
 82     createCanvas: function (width, height, tabIndex) {
 83         var canvas = this.createHTMLElement(rock.constants.CANVAS_TAG);
 84         canvas.setAttribute('width', width + 'px');
 85         canvas.setAttribute('height', height + 'px');
 86         if (!rock.isNullOrUndefined(tabIndex)) {
 87             canvas.setAttribute('tabindex', tabIndex);
 88         }
 89         return canvas;
 90     }
 91 
 92 };
