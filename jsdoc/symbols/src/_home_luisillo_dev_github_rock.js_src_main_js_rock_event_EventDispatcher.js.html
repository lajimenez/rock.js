  3 /**
  4  * Base class for components that can dispatch events.
  5  *
  6  * SOME IDEAS...
  7  * I think the event system could be implemented using the DOM 'EventTarget' interface (for example, adding
  8  * the events to the canvas that any rock application has). But I like this implementation because
  9  * I can assure that the javascript thread won't be disrupted (and so no partial drawing should occur)
 10  * when an event is dispatched.
 11  *
 12  * @constructor
 13  *
 14  * @author Luis Alberto Jim√©nez
 15  */
 16 rock.event.EventDispatcher = function () {
 17     this.handlersEventElements = [];
 18 };
 19 
 20 /**
 21  * Dispatch an event
 22  *
 23  * @param {rock.event.Event} event
 24  *            the event
 25  */
 26 rock.event.EventDispatcher.prototype.dispatchEvent = function (event) {
 27     event.setSource(this);
 28     var currentHandlersEventElement = this.getHandlersEventElement(event
 29         .getId());
 30     var i;
 31     var handlers;
 32     if (currentHandlersEventElement != null) {
 33         handlers = currentHandlersEventElement.handlers;
 34         for (i = 0; i < handlers.length; i++) {
 35             handlers[i](event);
 36         }
 37 
 38     }
 39 };
 40 
 41 /**
 42  * Add a new event listener to an event type (defined by id)
 43  *
 44  * @param eventId
 45  *            the id of the event to listen
 46  * @param {function} handler
 47  *            function to handle the event
 48  */
 49 rock.event.EventDispatcher.prototype.addEventListener = function (eventId, handler) {
 50     var currentHandlersEventElement = this.getHandlersEventElement(eventId);
 51 
 52     if (currentHandlersEventElement == null) {
 53         currentHandlersEventElement = new rock.event.EventDispatcherHandlersEventElement();
 54         currentHandlersEventElement.eventId = eventId;
 55         this.handlersEventElements.push(currentHandlersEventElement);
 56     }
 57 
 58     currentHandlersEventElement.handlers.push(handler);
 59 };
 60 
 61 /**
 62  * Remove an event listener from an event type (defined by id)
 63  *
 64  * @param eventId
 65  *            the id of the event to remove
 66  * @param handler
 67  *            function to handle the event {function}
 68  */
 69 rock.event.EventDispatcher.prototype.removeEventListener = function (eventId, handler) {
 70     var currentHandlersEventElement = this.getHandlersEventElement(eventId);
 71 
 72     if (currentHandlersEventElement == null) {
 73         return;
 74     }
 75 
 76     rock.util.JsUtils.removeByValueFromArray(currentHandlersEventElement.handlers,
 77         handler);
 78 };
 79 
 80 /**
 81  * Remove all event listener from an event type (defined by id)
 82  *
 83  * @param eventId
 84  *            the id of the event to remove
 85  */
 86 rock.event.EventDispatcher.prototype.removeAllEventListener = function (eventId) {
 87     var currentHandlersEventElement = this.getHandlersEventElement(eventId);
 88 
 89     if (currentHandlersEventElement == null) {
 90         return;
 91     }
 92 
 93     rock.util.JsUtils.clearArray(currentHandlersEventElement.handlers);
 94 };
 95 
 96 /**
 97  * Get the handlers for an event type
 98  *
 99  * @param eventId
100  *            the event id
101  * @returns list of handlers
102  * @private
103  */
104 rock.event.EventDispatcher.prototype.getHandlersEventElement = function (eventId) {
105     var i;
106     var handlersEventElements = this.handlersEventElements;
107     var currentElement;
108 
109     for (i = 0; i < handlersEventElements.length; i++) {
110         currentElement = handlersEventElements[i];
111 
112         if (eventId === currentElement.eventId) {
113             return currentElement;
114         }
115     }
116 
117     return null;
118 };

120 /**
121  * Internal class to store relation between events and handlers
122  */
123 rock.event.EventDispatcherHandlersEventElement = function () {
124     this.eventId = null;
125     this.handlers = [];
126 };
