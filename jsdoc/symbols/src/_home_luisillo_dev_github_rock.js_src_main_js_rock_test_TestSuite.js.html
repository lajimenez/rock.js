  3 /**
  4  * Abstract class for implementing tests. The tests must be prefixed with 'test'.
  5  *
  6  * @constructor
  7  * @abstract
  8  *
  9  * @author Luis Alberto Jim√©nez
 10  */
 11 rock.test.TestSuite = function () {
 12     this.currentTestFailed = null;
 13     this.reset();
 14 };
 15 
 16 /**
 17  * Reset the test so it can be run again
 18  */
 19 rock.test.TestSuite.prototype.reset = function () {
 20     this.currentTestFailed = false;
 21 };
 22 
 23 /**
 24  * Returns if the test has failed
 25  *
 26  * @returns {boolean} if the test has failed
 27  */
 28 rock.test.TestSuite.prototype.hasCurrentTestFailed = function () {
 29     return this.currentTestFailed;
 30 };
 31 
 32 /**
 33  * Any initialization needed must be done here
 34  */
 35 rock.test.TestSuite.prototype.setUp = function () {
 36 };
 37 
 38 /**
 39  * Any release of resources must be done here
 40  */
 41 rock.test.TestSuite.prototype.tearDown = function () {
 42 };

 44 /**
 45  * Assert if the param is true
 46  *
 47  * @param value
 48  *      the value to be checked
 49  */
 50 rock.test.TestSuite.prototype.assertTrue = function (value) {
 51     if (!value) {
 52         this.markCurrentTestAsFailed();
 53     }
 54 };
 55 
 56 /**
 57  * Assert if the param is false
 58  *
 59  * @param value
 60  *      the value to be checked
 61  */
 62 rock.test.TestSuite.prototype.assertFalse = function (value) {
 63     if (value) {
 64         this.markCurrentTestAsFailed();
 65     }
 66 };
 67 
 68 /**
 69  * Assert if both params are equals
 70  *
 71  * @param expectation
 72  *      the expectation value
 73  * @param value
 74  *      the value to be checked
 75  */
 76 rock.test.TestSuite.prototype.assertEquals = function (expectation, value) {
 77     if (expectation !== value) {
 78         this.markCurrentTestAsFailed();
 79     }
 80 };
 81 
 82 /**
 83  * Assert if params are different
 84  *
 85  * @param expectation
 86  *      the expectation value
 87  * @param value
 88  *      the value to be checked
 89  */
 90 rock.test.TestSuite.prototype.assertNotEquals = function (expectation, value) {
 91     if (expectation === value) {
 92         this.markCurrentTestAsFailed();
 93     }
 94 };
 95 
 96 /**
 97  * Assert if value is null
 98 
 99  * @param value
100  *      the value to be checked
101  */
102 rock.test.TestSuite.prototype.assertNull = function (value) {
103     if (value !== null) {
104         this.markCurrentTestAsFailed();
105     }
106 };
107 
108 /**
109  * Assert if value is null or undefined
110 
111  * @param value
112  *      the value to be checked
113  */
114 rock.test.TestSuite.prototype.assertNullOrUndefined = function (value) {
115     if (!rock.util.JsUtils.isNullOrUndefined(value)) {
116         this.markCurrentTestAsFailed();
117     }
118 };
119 
120 /**
121  * Assert if value is not null
122 
123  * @param value
124  *      the value to be checked
125  */
126 rock.test.TestSuite.prototype.assertNotNull = function (value) {
127     if (value === null) {
128         this.markCurrentTestAsFailed();
129     }
130 };
131 
132 /**
133  * Assert that function throws error
134  *
135  * @param expectedMessage
136  * @param object
137  * @param func
138  * @param params
139  */
140 rock.test.TestSuite.prototype.assertError = function (expectedMessage, object, func, params) {
141     var error = null;
142     try {
143         func.apply(object, params);
144     } catch (e) {
145         error = e;
146     }
147 
148     if (!error) {
149         this.markCurrentTestAsFailed();
150     } else {
151         if (expectedMessage != error.message) {
152             this.markCurrentTestAsFailed();
153         }
154     }
155 };
156 
157 /**
158  * Mark the current test as failed
159  */
160 rock.test.TestSuite.prototype.markCurrentTestAsFailed = function () {
161     this.currentTestFailed = true;
