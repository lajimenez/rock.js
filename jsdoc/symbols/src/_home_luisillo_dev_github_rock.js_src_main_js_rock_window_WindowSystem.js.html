  3 /**
  4  * Base class for window system. Handles the current window to draw.
  5  *
  6  * @param {rock.app.Application} application
  7  *            the application where this window system will be part
  8  *
  9  * @constructor
 10  * @author Luis Alberto Jim√©nez
 11  */
 12 rock.window.WindowSystem = function (application) {
 13     this.application = application;
 14     this.canvas = application.getCanvas();
 15     this.width = -1;
 16     this.height = -1;
 17     this.graphicsEngine = null;
 18     this.registeredWindowsInfo = [];
 19     this.currentWindow = null;
 20     this.HTMLlisteners = {};
 21     this.canDraw = true;
 22 
 23     // Handle context events
 24     application.addEventListener(rock.constants.ROCK_EVENT_CONTEXT_LOST,
 25         rock.createEventHandler(this, this.disableDrawing));
 26     application.addEventListener(rock.constants.ROCK_EVENT_CONTEXT_RESTORED,
 27         rock.createEventHandler(this, this.enableDrawing));
 28 
 29     application.addEventListener(rock.app.Application.LOCALE_CHANGED_EVENT,
 30         rock.createEventHandler(this,this.handleOnLocaleChanged));
 31 };

 33 /**
 34  * Create a new MouseEvent
 35  *
 36  * @param eventId
 37  *            id of the event to be thrown
 38  * @param x
 39  *            the x click
 40  * @param y
 41  *            the y click
 42  * @param DOMMouseEvent
 43  *            the event thrown by the DOM
 44  * @returns {rock.event.MouseEvent} a new Mouse Event
 45  *
 46  */
 47 rock.window.WindowSystem.createMouseEvent = function (eventId, x, y, DOMMouseEvent) {
 48     var mouseEvent = new rock.event.MouseEvent(eventId);
 49     mouseEvent.setX(x);
 50     mouseEvent.setY(y);
 51     mouseEvent.setDOMMouseEvent(DOMMouseEvent);
 52     return mouseEvent;
 53 };

 80 /**
 81  * Add an HTML event listener
 82  *
 83  * @param {String} eventType
 84  * @param {function} listener
 85  *
 86  * @private
 87  */
 88 rock.window.WindowSystem.prototype.addHTMLEventListener = function (eventType, listener) {
 89     this.removeHTMLEventListener(eventType);
 90     rock.util.DOMUtils.addEventListener(this.canvas, eventType, listener);
 91     this.HTMLlisteners[eventType] = listener;
 92 };

 94 /**
 95  * Remove the HTML event listener currently used
 96  *
 97  * @param {String} eventType
 98  *
 99  * @private
100  */
101 rock.window.WindowSystem.prototype.removeHTMLEventListener = function (eventType) {
102     var listener = this.HTMLlisteners[eventType];
103     if (rock.util.JsUtils.isNullOrUndefined(listener)) {
104         return;
105     }
106 
107     rock.util.DOMUtils.removeEventListener(this.canvas, eventType, listener);
108     this.HTMLlisteners[eventType] = null;
109 };

160 /**
161  * Remove all HTML events listeners. You can do that if you want to handle events by yourself.
162  * (It could be nice if you don't want to use default behavior that creates a lot of objects
163  * and so, a lot of memory)
164  */
165 rock.window.WindowSystem.prototype.removeHTMLEventsListeners = function () {
166     var HTMLlisteners = this.HTMLlisteners;
167     var property;
168     for (property in HTMLlisteners) {
169         if (HTMLlisteners.hasOwnProperty(property)) {
170             this.removeHTMLEventListener(HTMLlisteners[property]);
171         }
172     }
173 };

254 /**
255  * Handle the mouse leave in the window
256  */
257 rock.window.WindowSystem.prototype.handleMouseLeave = function (event) {
258     var coords = rock.util.MouseUtils.resolveEventCoords(event);
259     var x = coords.x;
260     var y = coords.y;
261 
262     var i;
263     var currentComponent;
264     var components = this.currentWindow.getComponents();
265     for (i = 0; i < components.length; i++) {
266         currentComponent = components[i];
267         currentComponent.handleMouseOutMouseOverEvent(false, x, y, event);
268     }
269 
270     this.currentWindow.dispatchEvent(
271         rock.window.WindowSystem.createMouseEvent(rock.constants.ROCK_EVENT_MOUSE_LEAVE, x, y, event));
272 };

303  * Handle a system mouse event
304  *
305  * @param {rock.window.Window} window
306  *            the window handling the event
307  * @param eventId
308  *            id of the event to be thrown
309  * @param mouseEvent
310  *            the event thrown by the DOM
311  * @private
312  */
313 rock.window.WindowSystem.prototype.handleMouseSystemEvent = function (window, eventId, mouseEvent) {
314     var coords = rock.util.MouseUtils.resolveEventCoords(mouseEvent);
315     var x = coords.x;
316     var y = coords.y;
317 
318     var i;
319     var currentComponent;
320     var components = window.getComponents();
321     for (i = 0; i < components.length; i++) {
322         currentComponent = components[i];
323         currentComponent.handleMouseSystemEvent(eventId, x, y, mouseEvent);
324     }
325 
326     window.dispatchEvent(rock.window.WindowSystem.createMouseEvent(eventId, x, y, mouseEvent));
327 };

386 /**
387  * Get a window in the system
388  *
389  * @param id
390  *            the id of the window
391  * @returns {rock.window.Window} the window
392  */
393 rock.window.WindowSystem.prototype.getWindow = function (id) {
394     var registeredWindowInfo = this.getRegisteredWindowInfo(id);
395     if (registeredWindowInfo != null) {
396         return registeredWindowInfo.getWindow();
397     } else {
398         return null;
399     }
400 };

402 /**
403  * Get a window registered info in the system
404  *
405  * @param id
406  *            the id of the window
407  * @returns {rock.window.WindowRegisterInfo} the window
408  * @private
409  */
410 rock.window.WindowSystem.prototype.getRegisteredWindowInfo = function (id) {
411     var registeredWindowsInfo = this.registeredWindowsInfo;
412     var registeredWindowInfo = null;
413     var i;
414     for (i = 0; i < registeredWindowsInfo.length; i++) {
415         registeredWindowInfo = registeredWindowsInfo[i];
416         if (registeredWindowInfo.getId() === id) {
417             return registeredWindowInfo;
418         }
419     }
420 
421     return null;
422 };

424 /**
425  * Sets the current window.
426  *
427  * @param {rock.window.Window} currentWindow
428  *            the current window to draw
429  */
430 rock.window.WindowSystem.prototype.setCurrentWindow = function (currentWindow) {
431     var windowToDeactivate = this.currentWindow;
432     if (!rock.isNullOrUndefined(windowToDeactivate)) {
433         windowToDeactivate.deactivate();
434     }
435 
436     this.currentWindow = currentWindow;
437     this.currentWindow.dispatchEvent(new rock.event.Event(rock.window.Window.WINDOW_ACTIVATED_EVENT));
438     this.draw();
439 };
