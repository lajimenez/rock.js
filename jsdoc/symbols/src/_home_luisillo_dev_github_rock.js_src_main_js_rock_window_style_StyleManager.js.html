  3 /**
  4  * Class for managing windows and components styles.
  5  * Use this class for initializing styles, not for dynamically changing it.
  6  *
  7  * @constructor
  8  *
  9  * @author Luis Alberto Jim√©nez
 10  */
 11 rock.window.style.StyleManager = function () {
 12     this.styles = [];
 13     this.currentStyle = null;
 14 
 15     var rockStyle = rock.window.style.StyleManager.createRockStyle();
 16     this.addStyle(rockStyle);
 17     this.setCurrentStyle(rockStyle.getId());
 18 };
 19 
 20 rock.window.style.StyleManager.WINDOW_BACKGROUND_COLOR = 'WINDOW_BACKGROUND_COLOR';
 21 
 22 rock.window.style.StyleManager.COMPONENT_BACKGROUND_COLOR = 'COMPONENT_BACKGROUND_COLOR';
 23 
 24 rock.window.style.StyleManager.COMPONENT_COLOR = 'COMPONENT_COLOR';
 25 
 26 rock.window.style.StyleManager.COMPONENT_COLOR_2 = 'COMPONENT_COLOR_2';
 27 
 28 rock.window.style.StyleManager.COMPONENT_FONT_COLOR = 'COMPONENT_FONT_COLOR';
 29 
 30 rock.window.style.StyleManager.COMPONENT_FONT = 'COMPONENT_FONT';
 31 
 32 /**
 33  * Create the default rock style
 34  *
 35  * @returns {rock.window.style.Style}
 36  */
 37 rock.window.style.StyleManager.createRockStyle = function () {
 38     var ROCK_STYLE_ID = rock.constants.ROCK_STYLE_ID;
 39     var rockStyle = new rock.window.style.Style(ROCK_STYLE_ID);
 40 
 41     rockStyle.addColor(rock.window.style.StyleManager.WINDOW_BACKGROUND_COLOR,
 42         new rock.graphics.Color(174, 222, 255));
 43 
 44     rockStyle.addColor(rock.window.style.StyleManager.COMPONENT_BACKGROUND_COLOR,
 45         new rock.graphics.Color(123, 201, 255));
 46 
 47     rockStyle.addColor(rock.window.style.StyleManager.COMPONENT_COLOR,
 48         new rock.graphics.Color(20, 30, 128));
 49 
 50     rockStyle.addColor(rock.window.style.StyleManager.COMPONENT_COLOR_2,
 51         new rock.graphics.Color(123, 175, 255));
 52 
 53     rockStyle.addColor(rock.window.style.StyleManager.COMPONENT_FONT_COLOR,
 54         new rock.graphics.Color(0, 0, 0));
 55 
 56     rockStyle.addFont(rock.window.style.StyleManager.COMPONENT_FONT,
 57         new rock.graphics.Font(rock.constants.DEFAULT_FONT_TYPE, rock.constants.DEFAULT_FONT_SIZE));
 58 
 59     return rockStyle;
 60 };

 62 /**
 63  * Add a new style
 64  *
 65  * @param {rock.window.style.Style} style
 66  *      the style to add
 67  */
 68 rock.window.style.StyleManager.prototype.addStyle = function (style) {
 69     this.styles.push(style);
 70 };
 71 
 72 /**
 73  * Set the current style. Any call to get font or color will use it
 74  *
 75  * @param {String} id
 76  *      the id of the style to set
 77  */
 78 rock.window.style.StyleManager.prototype.setCurrentStyle = function (id) {
 79     var i, style;
 80     var styles = this.styles;
 81     for (i = 0; i < styles.length; i++) {
 82         style = styles[i];
 83         if (style.getId() == id) {
 84             this.currentStyle = style;
 85         }
 86     }
 87 };
 88 
 89 /**
 90  * Get a color form the current style
 91  *
 92  * @param {String} colorId
 93  *      the id of the color
 94  * @returns {rock.graphics.Color}
 95  */
 96 rock.window.style.StyleManager.prototype.getColor = function (colorId) {
 97     var color  = this.currentStyle.getColor(colorId);
 98     if (!rock.util.JsUtils.isNullOrUndefined(color)) {
 99         return color.clone();
100     } else {
101         return null;
102     }
103 };
104 
105 /**
106  * Get a font form the current style
107  *
108  * @param {String} fontId
109  *      the id of the font
110  * @returns {rock.graphics.Font}
111  */
112 rock.window.style.StyleManager.prototype.getFont = function (fontId) {
113     var font  = this.currentStyle.getFont(fontId);
114     if (!rock.util.JsUtils.isNullOrUndefined(font)) {
115         return font.clone();
116     } else {
117         return null;
118     }
