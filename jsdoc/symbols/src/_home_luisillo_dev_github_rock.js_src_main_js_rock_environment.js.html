1 /**
  2  * Represents the environment where rock is running.
  3  *
  4  * @author Luis Alberto Jim√©nez

  6 rock.environment = {
  7 
  8     /**
  9      * User agent text (from navigator)
 10      */
 11     userAgent: navigator.userAgent,
 12 
 13     /**
 14      * User agent family
 15      */
 16     userAgentFamily: null,
 17 
 18     updateUserAgentFamily: function() {
 19         var userAgent = this.userAgent;
 20         if (rock.stringContains(userAgent, rock.constants.USER_AGENT_NAVIGATOR_TEXT_CHROME)) {
 21             this.userAgentFamily = rock.constants.USER_AGENT_FAMILY_CHROME;
 22         } else if (rock.stringContains(userAgent, rock.constants.USER_AGENT_NAVIGATOR_TEXT_FIREFOX)) {
 23             this.userAgentFamily = rock.constants.USER_AGENT_FAMILY_FIREFOX;
 24         } else if (rock.stringContains(userAgent, rock.constants.USER_AGENT_NAVIGATOR_TEXT_INTERNET_EXPLORER)) {
 25             this.userAgentFamily = rock.constants.USER_AGENT_FAMILY_INTERNET_EXPLORER;
 26         } else if (rock.stringContains(userAgent, rock.constants.USER_AGENT_NAVIGATOR_TEXT_INTERNET_EXPLORER_TRIDENT)) {
 27             // IE 11 (Revise it...)
 28             this.userAgentFamily = rock.constants.USER_AGENT_FAMILY_INTERNET_EXPLORER;
 29         }
 30     }
 31 };
 32 
 33 rock.environment.updateUserAgentFamily();
