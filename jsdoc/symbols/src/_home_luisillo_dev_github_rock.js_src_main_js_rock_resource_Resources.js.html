  3 /**
  4  * Class for store resources
  5  *
  6  * @constructor
  7  * @author Luis Alberto Jim√©nez

  9 rock.resource.Resources = function (locale) {
 10     this.locale = locale;
 11     if (rock.util.JsUtils.isNullOrUndefined(this.locale)) {
 12         this.locale = rock.constants.DEFAULT_ROCK_LOCALE;
 13     }
 14 
 15     this.resources = {};
 16 };
 17 
 18 rock.resource.Resources.prototype.getResourcesByType = function (type) {
 19     this.resources[type] = this.resources[type] || {};
 20     return this.resources[type];
 21 };
 22 
 23 /**
 24  * Add a new resource
 25  *
 26  * @param type
 27  *      the type of the resource
 28  * @param id
 29  *      the id of the resource
 30  * @param {object} resource
 31  *      the resource itself
 32  */
 33 rock.resource.Resources.prototype.addResource = function (type, id, resource) {
 34     var resources = this.getResourcesByType(type);
 35     resources[id] = resource;
 36 };
 37 
 38 /**
 39  * Get a resource
 40  *
 41  * @param type
 42  *      the type of the resource
 43  * @param id
 44  *      the id of the resource
 45  * @returns {object} the resource
 46  */
 47 rock.resource.Resources.prototype.getResource = function (type, id) {
 48     var resources = this.getResourcesByType(type);
 49     return resources[id];
 50 };
