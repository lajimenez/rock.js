  3 /**
  4  * Represents a 3x3 matrix.
  5  *
  6  * @constructor
  7  * @author Luis Alberto Jim√©nez
  8  */
  9 rock.geometry.Matrix3 = function () {
 10     // column major
 11     this.matrix = new Array(9);
 12     this.identity();
 13 
 14     // This properties are used to avoid allocate new memory
 15     this.point3_ma = new rock.geometry.Point3(0, 0, 0);
 16 };
 17 
 18 rock.geometry.Matrix3.prototype.getAuxiliaryPoint3 = function () {
 19     var point3 = this.point3_ma;
 20     point3.setX(0);
 21     point3.setY(0);
 22     point3.setZ(0);
 23     return point3;
 24 };
 25 
 26 /**
 27  * Returns the matrix as an array
 28  *
 29  * @returns {Array.9}
 30  */
 31 rock.geometry.Matrix3.prototype.getMatrixAsArray = function () {
 32     return this.matrix;
 33 };
 34 
 35 /**
 36  * Do this matrix as identity matrix
 37  *
 38  */
 39 rock.geometry.Matrix3.prototype.identity = function () {
 40     for (var i = 0; i < 9; i++) {
 41         this.matrix[i] = 0;
 42     }
 43 
 44     this.matrix[0] = 1; // this.setValue(0, 0, 1);
 45     this.matrix[4] = 1; // this.setValue(1, 1, 1);
 46     this.matrix[8] = 1; // this.setValue(2, 2, 1);
 47 };
 48 
 49 /**
 50  * Calculate the determinant
 51  *
 52  * @returns the determinant
 53  */
 54 rock.geometry.Matrix3.prototype.getDeterminant = function () {
 55     var determinant = ((this.matrix[0] * this.matrix[4] * this.matrix[8])
 56         + (this.matrix[3] * this.matrix[7] * this.matrix[2]) + (this.matrix[6]
 57         * this.matrix[1] * this.matrix[5]))
 58         - ((this.matrix[6] * this.matrix[4] * this.matrix[2])
 59         + (this.matrix[3] * this.matrix[1] * this.matrix[8]) + (this.matrix[0]
 60         * this.matrix[7] * this.matrix[5]));
 61     return determinant;
 62 };

 64 /**
 65  * This function multiply a point with a matrix
 66  *
 67  * @param {rock.geometry.Point3} point3
 68  *      the point
 69  *
 70  * @return {rock.geometry.Point3}
 71  */
 72 rock.geometry.Matrix3.prototype.multiplyByPoint3 = function (point3) {
 73     var result = this.getAuxiliaryPoint3();
 74 
 75     var point3X = point3.getX();
 76     var point3Y = point3.getY();
 77     var point3Z = point3.getZ();
 78 
 79     result.setX(point3X * this.getValue(0, 0) + point3Y * this.getValue(0, 1) + point3Z * this.getValue(0, 2));
 80     result.setY(point3X * this.getValue(1, 0) + point3Y * this.getValue(1, 1) + point3Z * this.getValue(1, 2));
 81     result.setZ(point3X * this.getValue(2, 0) + point3Y * this.getValue(2, 1) + point3Z * this.getValue(2, 2));
 82 
 83     return result;
 84 };

 86 /**
 87  * Set a value
 88  *
 89  * @param i
 90  *            the row
 91  * @param j
 92  *            the column
 93  * @param value
 94  *            the value to set
 95  */
 96 rock.geometry.Matrix3.prototype.setValue = function (i, j, value) {
 97     var pos = j * 3 + i;
 98     this.matrix[pos] = value;
 99 };

101 /**
102  * Return a value
103  *
104  * @param i
105  *            the row
106  * @param j
107  *            the column
108  * @returns the value
109  */
110 rock.geometry.Matrix3.prototype.getValue = function (i, j) {
111     var pos = j * 3 + i;
112     return this.matrix[pos];
113 };
