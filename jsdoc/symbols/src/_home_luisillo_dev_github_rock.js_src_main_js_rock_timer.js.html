1 /**
  2  * Functions to handle timer operations
  3  *
  4  * @constructor
  5  * @author Luis Alberto Jim√©nez
  6  */
  7 rock.timer = {
  8 
  9     /**
 10      * Array containing timer functions
 11      */
 12     registeredFunctions : [],
 13 
 14     /**
 15      * You can use this property to check if the timer is running
 16      */
 17     isRunning : false,
 18 
 19     request : null,
 20 
 21     /**
 22      * Register a function so it will be executed when starting timer
 23      *
 24      * @param {Object} context
 25      *      the context which function will be executed
 26      * @param {function} func
 27      *      the function to be executed
 28      * @param {Array} params
 29      *      array containing parameters whe calling function
 30      */
 31     registerTimerFunction : function (context, func, params) {
 32         var functionInfo = {
 33             context : context,
 34             func : func,
 35             params : params
 36         };
 37         this.registeredFunctions.push(functionInfo);
 38     },
 39 
 40     /**
 41      * Unregister a function so it won't be executed when starting timer
 42      *
 43      * @param {function} func
 44      *      the function to be unregistered
 45      */
 46     unregisterTimerFunction : function (func) {
 47         var i, functionInfo, registeredFunctions;
 48         registeredFunctions = this.registeredFunctions;
 49         for (i = 0; i < registeredFunctions.length; i++) {
 50             functionInfo = registeredFunctions[i];
 51             if (func === functionInfo.func) {
 52                 rock.util.JsUtils.removeByValueFromArray(registeredFunctions, functionInfo);
 53             }
 54         }
 55     },
 56 
 57     /**
 58      * Start the timer
 59      */
 60     start : function () {
 61         // You can start the timer without any function registered
 62         if (!this.isRunning) {
 63             this.request = requestAnimationFrame(this.rockTimerHandle);
 64             this.isRunning = true;
 65         }
 66     },
 67 
 68     /**
 69      * Stop the timer
 70      */
 71     stop : function () {
 72         if (this.isRunning) {
 73             cancelAnimationFrame(this.request);
 74             this.isRunning = false;
 75         }
 76     },
 77 
 78     /**
 79      * Internal function to handle timer
 80      *
 81      * @private
 82      */
 83     rockTimerHandle : function () {
 84         var i, registeredFunctions, functionInfo;
 85         registeredFunctions = rock.timer.registeredFunctions;
 86         rock.timer.request = requestAnimationFrame(rock.timer.rockTimerHandle);
 87         for (i = 0; i < registeredFunctions.length; i++) {
 88             functionInfo = registeredFunctions[i];
 89             functionInfo.func.apply(functionInfo.context, functionInfo.params);
 90         }
 91     },
 92 
 93     /**
 94      * Rock implementation when standard 'requestAnimationFrame' is not available
 95      *
 96      * @param {function} func
 97      *      the function to be executed
 98      * @returns {number} id of the interval
 99      */
100     requestAnimationFrame : function (func) {
101         return setTimeout(func, 16);
102     },
103 
104     /**
105      * Rock implementation when standard 'cancelAnimationFrame' is not available
106      *
107      * @param requestID
108      *      id of the interval so clear
109      */
110     cancelAnimationFrame : function (requestID) {
111         clearTimeout(requestID);
112     }
113 
114 };
115

120         window.mozRequestAnimationFrame || window.oRequestAnimationFrame ||
121         window.msRequestAnimationFrame || rock.timer.requestAnimationFrame;
122 
123     window.cancelAnimationFrame =
124         window.cancelAnimationFrame || window.webkitCancelRequestAnimationFrame ||
125         window.mozCancelRequestAnimationFrame || window.oCancelRequestAnimationFrame ||
126         window.msCancelRequestAnimationFrame || rock.timer.cancelAnimationFrame;
127 })();
