  9 rock.resource.ResourceManager = function () {
 10     // Use this locale when no other locale is required
 11     this.DEFAULT_LOCALE = rock.constants.DEFAULT_ROCK_LOCALE;
 12 
 13     this.locale = this.DEFAULT_LOCALE;
 14     this.resources = {};
 15     this.defaultResources = this.createResourceObject(this.DEFAULT_LOCALE);
 16 };
 17 
 18 rock.resource.ResourceManager.prototype.createResourceObject = function (locale) {
 19     if (rock.util.JsUtils.isNullOrUndefined(this.resources[locale])) {
 20         this.resources[locale] = new rock.resource.Resources(locale);
 21     }
 22 
 23     return this.resources[locale];
 24 };
 25 
 26 rock.resource.ResourceManager.prototype.resolveLocale = function (locale) {
 27     var resolvedLocale = locale;
 28     // If no locale requested, we use default
 29     if (rock.util.JsUtils.isNullOrUndefined(locale)) {
 30         resolvedLocale = this.DEFAULT_LOCALE;
 31     }
 32 
 33     return resolvedLocale;
 34 };
 35 
 36 /**
 37  * Get resources related to a locale
 38  *
 39  * @param locale
 40  *        the locale
 41  *
 42  * @returns {rock.resource.Resources}
 43  */
 44 rock.resource.ResourceManager.prototype.getResources = function (locale) {
 45     var resolvedLocale = this.resolveLocale(locale);
 46 
 47     var resources = this.resources[resolvedLocale];
 48     if (rock.util.JsUtils.isNullOrUndefined(resources)) {
 49         resources = this.createResourceObject(resolvedLocale);
 50     }
 51     return resources;
 52 };

 85 /**
 86  * Load all strings for a locale
 87  *
 88  * @param {Object} json
 89  *      JSON containing messages for locale
 90  */
 91 rock.resource.ResourceManager.prototype.loadStrings = function (json) {
 92     var locale = json.locale;
 93     var messages = json.messages;
 94     var property;
 95 
 96     for (property in messages) {
 97         if (messages.hasOwnProperty(property)) {
 98             this.addString(property, messages[property], locale);
 99         }
100     }
101 };

131 /**
132  * Get a resource
133  *
134  * @param type
135  *      the type of the resource
136  * @param id
137  *      the id of the resource
138  * @param {String} [locale]
139  *      the locale (if undefined, current locale will be used)
140  *
141  * @returns {object} the resource
142  */
143 rock.resource.ResourceManager.prototype.getResource = function (type, id, locale) {
144     var localeToUse = locale;
145     if (rock.util.JsUtils.isNullOrUndefined(localeToUse)) {
146         localeToUse = this.locale;
147     }
148 
149     var resources = this.getResources(localeToUse);
150     var resource = resources.getResource(type, id);
151 
152     // If not found, try to find it in default
153     if (rock.util.JsUtils.isNullOrUndefined(resource)) {
154         resource = this.defaultResources.getResource(type, id);
155     }
156 
157     return resource;
158 };

160 /**
161  * Returns an String by id
162  *
163  * @param {String} id
164  *      the id of the string
165  * @param {Array} [expressions]
166  *      expressions to be replaced
167  * @param {String} [locale]
168  *      the locale (if undefined, current locale will be used)
169  *
170  * @returns {String}
171  */
172 rock.resource.ResourceManager.prototype.getString = function (id, expressions, locale) {
173     var message = this.getResource(rock.constants.RESOURCE_TYPE_STRING, id, locale);
174     var exps = expressions;
175 
176     if (rock.util.JsUtils.isNullOrUndefined(message)) {
177         message = id;
178     }
179 
180     if (rock.util.JsUtils.isNullOrUndefined(exps)) {
181         exps = [];
182     }
183 
184     var i, exp;
185     for (i = 0; i < exps.length; i++) {
186         exp = '{' + i + '}';
187         message = message.replace(exp, exps[i]);
188     }
189     return message;
190 };
