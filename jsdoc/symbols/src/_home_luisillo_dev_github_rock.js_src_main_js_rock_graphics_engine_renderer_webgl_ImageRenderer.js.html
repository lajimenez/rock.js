  3 /**
  4  * Render an image
  5  *
  6  * @param {rock.graphics.engine.webgl.WebGLContext} glContext
  7  *         the WebGL context
  8  *
  9  * @constructor
 10  * @extends {rock.graphics.engine.renderer.webgl.GenericRenderer}
 11  *
 12  * @author Luis Alberto Jim√©nez
 13  */
 14 rock.graphics.engine.renderer.webgl.ImageRenderer = function (glContext) {
 15     this.ATTRIBUTE_VERTICES = 'a_vertices';
 16     this.ATTRIBUTE_TEXCOORD = 'a_texCoord';
 17     this.UNIFORM_RESOLUTION = 'u_resolution';
 18     this.UNIFORM_IMAGE = 'u_image';
 19 
 20     var attributes = [this.ATTRIBUTE_VERTICES, this.ATTRIBUTE_TEXCOORD];
 21     var uniforms = [this.UNIFORM_RESOLUTION];
 22     var textures = [this.UNIFORM_IMAGE];
 23 
 24     var superParams =[
 25         glContext,
 26         rock.graphics.engine.renderer.webgl.ImageRenderer.vertexShaderSrc,
 27         rock.graphics.engine.renderer.webgl.ImageRenderer.fragmentShaderSrc,
 28         attributes,
 29         uniforms,
 30         textures
 31     ];
 32     rock.super_(this, superParams);
 33 };
 34 
 35 rock.extends_(rock.graphics.engine.renderer.webgl.ImageRenderer, rock.graphics.engine.renderer.webgl.GenericRenderer);
 36 
 37 /**
 38  * @see rock.graphics.engine.renderer.webgl.GenericRenderer#populateAttributes
 39  * @override
 40  */
 41 rock.graphics.engine.renderer.webgl.ImageRenderer.prototype.populateAttributes = function (renderable) {
 42     var program = this.program;
 43     program.populateAttribute(this.ATTRIBUTE_VERTICES, renderable.getVertices(), 2);
 44     program.populateAttribute(this.ATTRIBUTE_TEXCOORD, renderable.getTextureCoordinates(), 2);
 45 };
 46 
 47 /**
 48  * @see rock.graphics.engine.renderer.webgl.GenericRenderer#populateTextures
 49  * @override
 50  */
 51 rock.graphics.engine.renderer.webgl.ImageRenderer.prototype.populateTextures = function (renderable) {
 52     var program = this.program;
 53     program.populateTexture(this.UNIFORM_IMAGE, renderable.getImage());
 54 };
 55 
 56 /**
 57  * @see rock.graphics.engine.renderer.webgl.GenericRenderer#setUniforms
 58  * @override
 59  */
 60 rock.graphics.engine.renderer.webgl.ImageRenderer.prototype.setUniforms = function (renderable) {
 61     var gl = this.glContext.getHTMLContext();
 62     var program = this.program;
 63     gl.uniform2f(program.getUniformLocation(this.UNIFORM_RESOLUTION), renderable.getWidth(), renderable.getHeight());
 64 };

 66 /**
 67  * Vertex shader source
 68  */
 69 rock.graphics.engine.renderer.webgl.ImageRenderer.vertexShaderSrc = ''
 70 + ' attribute vec2 a_vertices;                                           \n'
 71 + ' attribute vec2 a_texCoord;                                           \n'
 72 + ' uniform vec2 u_resolution;                                           \n'
 73 + ' varying vec2 v_texCoord;                                             \n'
 74 + ' void main() {                                                        \n'
 75 + '     vec2 zeroToOne = a_vertices / u_resolution;                      \n'
 76 + '     vec2 zeroToTwo = zeroToOne * 2.0;                                \n'
 77 + '     vec2 clipSpace = zeroToTwo - 1.0;                                \n'
 78 + '     gl_Position = vec4(clipSpace * vec2(1, -1), 0, 1);               \n'
 79 + '     v_texCoord = a_texCoord;                                         \n'
 80 + ' }                                                                    \n';
 81 
 82 /**
 83  * Fragment shader source
 84  */
 85 rock.graphics.engine.renderer.webgl.ImageRenderer.fragmentShaderSrc = ''
 86 + ' precision mediump float;                                             \n'
 87 + ' uniform sampler2D u_image;                                           \n'
 88 + ' varying vec2 v_texCoord;                                             \n'
 89 + ' void main() {                                                        \n'
 90 + '     gl_FragColor = texture2D(u_image, v_texCoord);                   \n'
 91 + ' }                                                                    \n';
