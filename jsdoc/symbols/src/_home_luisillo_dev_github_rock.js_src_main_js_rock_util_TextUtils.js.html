  3 /**
  4  * Text utils class
  5  *

  9 rock.util.TextUtils = {
 10     /**
 11      * Get the width for a given text
 12      *
 13      * @param text
 14      *      text to measure
 15      * @param {rock.graphics.Font} font
 16      *      the font
 17      * @returns {Number} the width
 18      */
 19     measureTextWidth: function (text, font) {
 20         this.context.font = font.toCanvasFormat();
 21         var metrics = this.context.measureText(text);
 22         return Math.ceil(metrics.width);
 23     },
 24 
 25     /**
 26      * Get the height for a given text
 27      *
 28      * ATTENTION: This implementation is stupid right now... But I have left because
 29      * in the future there will only be one place for a better implementation.
 30      *
 31      * ATTENTION 2: This value is an approximate...
 32      *
 33      * @param text
 34      *      text to measure
 35      * @param {rock.graphics.Font} font
 36      *      the font
 37      * @returns {Number} the width
 38      */
 39     measureTextHeight: function (text, font) {
 40         return Math.ceil(font.getSize() * 1.2);
 41     }
 42 };
 43 
 44 rock.util.TextUtils.context = rock.util.DOMUtils.createCanvas(1, 1).getContext(rock.constants.CANVAS_CONTEXT_2D);
