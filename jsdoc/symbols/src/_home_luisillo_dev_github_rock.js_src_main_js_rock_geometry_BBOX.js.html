 21 rock.geometry.BBOX = function (xMin, xMax, yMin, yMax, zMin, zMax) {
 22     this.xMin = xMin;
 23     this.xMax = xMax;
 24     this.yMin = yMin;
 25     this.yMax = yMax;
 26     this.zMin = zMin;
 27     this.zMax = zMax;
 28 
 29     this.center_ma = new rock.geometry.Point3(0, 0, 0);
 30 };

 32 /**
 33  * Return if the point intersects with the rectangle
 34  *
 35  * @param {rock.geometry.Point3} point
 36  *
 37  * @returns {boolean}
 38  */
 39 rock.geometry.BBOX.prototype.intersectWithPoint = function (point) {
 40     var pointX = point.getX();
 41     var pointY = point.getY();
 42     var pointZ = point.getZ();
 43 
 44     return this.xMin <= pointX && pointX <= this.xMax && this.yMin <= pointY && pointY <= this.yMax &&
 45         this.zMin <= pointZ && pointZ <= this.zMax;
 46 };

 48 /**
 49  * Return the center of the BBOX
 50  *
 51  * @return {rock.geometry.Point3} the center
 52  */
 53 rock.geometry.BBOX.prototype.getCenter = function () {
 54     var center = this.center_ma;
 55     center.setX((this.xMin + this.xMax) / 2);
 56     center.setY((this.yMin + this.yMax) / 2);
 57     center.setZ((this.zMin + this.zMax) / 2);
 58 
 59     return center;
 60 };

 62 /**
 63  * Update the size of this BBOX to also contain passed BBOX
 64  *
 65  * @param BBOX
 66  *      the bbox
 67  */
 68 rock.geometry.BBOX.prototype.join = function (BBOX) {
 69     var xMin = BBOX.getXMin();
 70     var xMax = BBOX.getXMax();
 71     var yMin = BBOX.getYMin();
 72     var yMax = BBOX.getYMax();
 73     var zMin = BBOX.getZMin();
 74     var zMax = BBOX.getZMax();
 75 
 76     if (xMin < this.xMin) {
 77         this.xMin = xMin;
 78     }
 79     if (xMax > this.xMax) {
 80         this.xMax = xMax;
 81     }
 82 
 83     if (yMin < this.yMin) {
 84         this.yMin = yMin;
 85     }
 86     if (yMax > this.yMax) {
 87         this.yMax = yMax;
 88     }
 89 
 90     if (zMin < this.zMin) {
 91         this.zMin = zMin;
 92     }
 93     if (zMax > this.zMax) {
 94         this.zMax = zMax;
 95     }
 96 };

218 /**
219  * Get this BBOX as if it was rotated. Be careful as this function return an approximated BBOX.
220  *
221  * @return {rock.geometry.BBOX} the BBOX  * Get this BBOX as if it was rotated
222  */
223 rock.geometry.BBOX.prototype.getRotatedBBOX = function (rotationX, rotationY, rotationZ) {
224     var i, pointX, pointY, pointZ;
225     var xMin = this.xMin, xMax = this.xMax, yMin = this.yMin, yMax = this.yMax, zMin = this.zMin, zMax = this.zMax;
226     var point = new rock.geometry.Point3(0, 0, 0);
227     var points = [];
228     var matrix = new rock.geometry.Matrix4();
229     matrix.identity();
230     matrix.rotateX(rotationX);
231     matrix.rotateY(rotationY);
232     matrix.rotateZ(rotationZ);
233 
234     point.setX(xMin);
235     point.setY(yMin);
236     point.setZ(zMin);
237     points.push(matrix.multiplyByPoint3(point).clone());
238 
239     point.setX(xMin);
240     point.setY(yMin);
241     point.setZ(zMax);
242     points.push(matrix.multiplyByPoint3(point).clone());
243 
244     point.setX(xMax);
245     point.setY(yMin);
246     point.setZ(zMin);
247     points.push(matrix.multiplyByPoint3(point).clone());
248 
249     point.setX(xMax);
250     point.setY(yMin);
251     point.setZ(zMax);
252     points.push(matrix.multiplyByPoint3(point).clone());
253 
254     point.setX(xMin);
255     point.setY(yMax);
256     point.setZ(zMin);
257     points.push(matrix.multiplyByPoint3(point).clone());
258 
259     point.setX(xMin);
260     point.setY(yMax);
261     point.setZ(zMax);
262     points.push(matrix.multiplyByPoint3(point).clone());
263 
264     point.setX(xMax);
265     point.setY(yMax);
266     point.setZ(zMin);
267     points.push(matrix.multiplyByPoint3(point).clone());
268 
269     point.setX(xMax);
270     point.setY(yMax);
271     point.setZ(zMax);
272     points.push(matrix.multiplyByPoint3(point).clone());
273 
274     point = points[0];
275     xMin = point.getX();
276     xMax = xMin;
277     yMin = point.getY();
278     yMax = yMin;
279     zMin = point.getZ();
280     zMax = zMin;
281     for (i = 1; i < points.length; i++) {
282         point = points[i];
283         pointX = point.getX();
284         pointY = point.getY();
285         pointZ = point.getZ();
286 
287         if (pointX < xMin) {
288             xMin = pointX;
289         }
290         if (pointX > xMax) {
291             xMax = pointX;
292         }
293         if (pointY < yMin) {
294             yMin = pointY;
295         }
296         if (pointY > yMax) {
297             yMax = pointY;
298         }
299         if (pointZ < zMin) {
300             zMin = pointZ;
301         }
302         if (pointZ > zMax) {
303             zMax = pointZ;
304         }
305     }
306 
307     return new rock.geometry.BBOX(xMin, xMax, yMin, yMax, zMin, zMax);
308 };
