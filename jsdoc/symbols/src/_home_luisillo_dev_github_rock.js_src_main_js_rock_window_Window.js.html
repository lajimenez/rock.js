  3 /**
  4  * Class for window. All classes to use in the WindowSystem must inherit it.
  5  *
  6  * @param {rock.window.WindowSystem} windowSystem
  7  *            the window system where this window will be part
  8  *
  9  * @constructor
 10  * @extends rock.event.EventDispatcher
 11  * @augments rock.graphics.engine.IDrawable
 12  *
 13  * @author Luis Alberto Jim√©nez
 14  */
 15 rock.window.Window = function (windowSystem) {
 16     rock.super_(this);
 17 
 18     this.windowSystem = windowSystem;
 19 
 20     // We check explicit to avoid bad initializations.
 21     // For example, create the window before create the window system.
 22     if (rock.isNullOrUndefined(this.windowSystem)) {
 23         throw new rock.error.RockError(rock.resource.rockResourceManager.getString('NO_WINDOW_SYSTEM'));
 24     }
 25 
 26     // For more efficiency, we put the application in property
 27     this.application = this.windowSystem.getApplication();
 28 
 29     /* list of components */
 30     this.components = [];
 31 
 32     this.backgroundColor =
 33         this.application.getStyleManager().getColor(rock.window.style.StyleManager.WINDOW_BACKGROUND_COLOR);
 34 };
 35 
 36 rock.extends_(rock.window.Window, rock.event.EventDispatcher);
 37 
 38 rock.implements_(rock.window.Window, rock.graphics.engine.IDrawable);

 54 /**
 55  * Add a component to the components list
 56  *
 57  * @param {rock.window.component.Component} component
 58  *            the component to add
 59  */
 60 rock.window.Window.prototype.addComponent = function (component) {
 61     //component.updateWindow(this);
 62     this.components.push(component);
 63 };

 65 /**
 66  * Remove a component to the components list
 67  *
 68  * @param {rock.window.component.Component} component
 69  *            the component to remove
 70  */
 71 rock.window.Window.prototype.removeComponent = function (component) {
 72     //component.updateWindow(null);
 73     rock.util.JsUtils.removeByValueFromArray(this.components, component);
 74 };

 83 /**
 84  * @override
 85  * @see rock.graphics.engine.IDrawable#draw
 86  */
 87 rock.window.Window.prototype.draw = function (graphicsEngine) {
 88     graphicsEngine.clear(this.backgroundColor);
 89     this.drawUI(graphicsEngine);
 90 };
 91 
 92 /**
 93  * Draw the user interface
 94  * @protected
 95  */
 96 rock.window.Window.prototype.drawUI = function (graphicsEngine) {
 97     var i;
 98     var components = this.components;
 99     for (i = 0; i < components.length; i++) {
100         components[i].draw(graphicsEngine);
101     }
102 };
103 
104 /**
105  * Redraw the window
106  * @protected
107  */
108 rock.window.Window.prototype.redraw = function () {
109     var windowSystem = this.windowSystem;
110     // If the windows we want to redraw is not current,
111     // we don't have to redraw
112     if (this != windowSystem.getCurrentWindow()) {
113         return;
114     }
115 
116     // We 'centralize' window drawing in window system
117     windowSystem.draw();
118 };
119 
120 /**
121  * Deactivate the window. This will do all necessary things when the window is not active
122  */
123 rock.window.Window.prototype.deactivate = function () {
124     var i;
125     var component, components = this.components;
126     for (i = 0; i < components.length; i++) {
127         component = components[i];
128         component.deactivate();
129     }
130     this.dispatchEvent(new rock.event.Event(rock.window.Window.WINDOW_DEACTIVATED_EVENT));
131 };

133 /**
134  * Get a component by id
135  *
136  * @param id
137  *      id of the component
138  *
139  * @returns {rock.window.component.Component}
140  */
141 rock.window.Window.prototype.getComponent = function (id) {
142     var components = this.components;
143     var i, component;
144     for (i = 0; i < components.length; i++) {
145         component = components[i];
146         if (id === component.getId()) {
147             return component;
148         }
149     }
150 
151     return null;
152 };
