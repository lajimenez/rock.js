 22             if (array[i] === val) {

105     /**
106      * Check if a string starts ends a prefix
107      *
108      * @param value
109      * @param suffix
110      * @returns {boolean}
111      */
112     stringEndsWith: function(value, suffix) {
113         if (this.isNullOrUndefined(value) || this.isNullOrUndefined(suffix)) {
114             return false;
115         }
116 
117         var lastIndex = value.lastIndexOf(suffix);
118         if (lastIndex < 0) {
119             return false;
120         }
121 
122         return lastIndex  == (value.length - suffix.length);
123     },
