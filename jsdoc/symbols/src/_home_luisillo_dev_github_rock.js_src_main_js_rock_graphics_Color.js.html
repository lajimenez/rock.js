  3 /**
  4  * Class to represents a color
  5  *
  6  * @param r
  7  *            red component {0, 255}
  8  * @param g
  9  *            green component {0, 255}
 10  * @param b
 11  *            blue component {0, 255}
 12  * @param [a]
 13  *            alpha component {0, 255}
 14  * @constructor
 15  * @author Luis Alberto Jim√©nez
 16  */
 17 rock.graphics.Color = function (r, g, b, a) {
 18 
 19     this.MAX_VALUE = 255;
 20 
 21     this.red = r;
 22     this.green = g;
 23     this.blue = b;
 24 
 25     if (!rock.isNullOrUndefined(a)) {
 26         // Alpha
 27         this.alpha = a;
 28     } else {
 29         this.alpha = this.MAX_VALUE;
 30     }
 31 
 32 };

210 /**
211  * Create an color object from an hex values
212  *
213  * @param hex
214  *      the color as hex. Format: #RRGGBB
215  * @returns {rock.graphics.Color} the color
216  */
217 rock.graphics.Color.fromHex = function (hex) {
218     var red = parseInt(hex.substring(1, 3), 16);
219     var green = parseInt(hex.substring(3, 5), 16);
220     var blue = parseInt(hex.substring(5, 7), 16);
221 
222     return new rock.graphics.Color(red, green, blue);
223 };
