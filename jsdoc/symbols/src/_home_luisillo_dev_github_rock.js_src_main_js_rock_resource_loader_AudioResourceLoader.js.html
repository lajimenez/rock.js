 10 rock.resource.loader.AudioResourceLoader = function () {
 11     rock.super_(this, arguments);
 12 };
 13 
 14 rock.extends_(rock.resource.loader.AudioResourceLoader,
 15     rock.resource.loader.AbstractResourceLoader);
 16 
 17 /**
 18  * @override
 19  * @see rock.resource.loader.AbstractResourceLoader#loadResource
 20  */
 21 rock.resource.loader.AudioResourceLoader.prototype.loadResource = function (elem) {
 22 
 23     var HTMLAudio = new Audio();
 24     HTMLAudio.id = elem.id;
 25 
 26     HTMLAudio.addEventListener('loadeddata', rock.createEventHandler(this,
 27         this.onLoadAudio, [elem, HTMLAudio]), false);
 28 
 29     HTMLAudio.onerror = rock.createEventHandler(this, this.onErrorAudio, [
 30         elem, HTMLAudio]);
 31 
 32     HTMLAudio.onabort = rock.createEventHandler(this, this.onAbortAudio, [
 33         elem, HTMLAudio]);
 34 
 35     HTMLAudio.src = elem.getHTTPParams().getUrl();
 36 
 37 };
 38 
 39 /*
 40  * Event handler for an audio
 41  */
 42 rock.resource.loader.AudioResourceLoader.prototype.onLoadAudio = function (event, elem, HTMLAudio) {
 43     this.callResourceLoader(event, elem, HTMLAudio, true);
 44 };
 45 
 46 /*
 47  * Event handler for an audio
 48  */
 49 rock.resource.loader.AudioResourceLoader.prototype.onErrorAudio = function (event, elem, HTMLAudio) {
 50     this.callResourceLoader(event, elem, HTMLAudio, false);
 51 };
 52 
 53 /*
 54  * Event handler for an audio
 55  */
 56 rock.resource.loader.AudioResourceLoader.prototype.onAbortAudio = function (event, elem, HTMLAudio) {
 57     this.callResourceLoader(event, elem, HTMLAudio, false);
 58 };
 59 
 60 rock.resource.loader.AudioResourceLoader.prototype.callResourceLoader = function (event, elem, HTMLAudio, success) {
 61     var audio = new rock.audio.Audio(HTMLAudio);
 62     this.resourceLoaderManager.onResourceLoadEvent(success, elem, audio);
 63 };
