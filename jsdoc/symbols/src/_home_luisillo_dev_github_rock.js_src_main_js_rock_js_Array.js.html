  3 /**
  4  * Rock custom Array implementation. This class is intended for a more controlled dynamic resize of javascript Arrays.
  5  * The objective is to avoid create new Arrays. You can use this class and rehuse Arrays without allocating memory
  6  * (at least you can have more control on how Array grow...)
  7  * This class is intended for being used this way:
  8  *  Create Array
  9  *  Populate Array
 10  *  Clear Array
 11  *  Go to 'Populate Array'
 12  *
 13  * @param [initialCapacity]
 14  *            the initial capacity of the array
 15  *
 16  * @param [resizeIncrement]
 17  *            if the array must be resized, the capacity will be augmented by this amount
 18  *
 19  * @constructor
 20  *
 21  * @author Luis Alberto Jim√©nez
 22  */
 23 rock.js.Array = function (initialCapacity, resizeIncrement) {
 24     this.length = 0;
 25     var capacity = rock.js.Array.DEFAULT_INITIAL_CAPACITY;
 26     if (!rock.util.JsUtils.isNullOrUndefined(initialCapacity) && initialCapacity > 0) {
 27         capacity = initialCapacity;
 28     }
 29     this.values = new Array(capacity);
 30     this.capacity = capacity;
 31 
 32     this.resizeIncrement = rock.js.Array.DEFAULT_RESIZE_INCREMENT;
 33     if (!rock.util.JsUtils.isNullOrUndefined(resizeIncrement) && resizeIncrement > 0) {
 34         this.resizeIncrement = resizeIncrement;
 35     }
 36 };
 37 
 38 rock.js.Array.DEFAULT_INITIAL_CAPACITY = 10;
 39 
 40 rock.js.Array.DEFAULT_RESIZE_INCREMENT = 5;
 41 
 42 /**
 43  * Resize the values array
 44  *
 45  * @private
 46  */
 47 rock.js.Array.prototype.resizeValues = function () {
 48     var resizeIncrement = this.resizeIncrement;
 49     var values = this.values;
 50     var i;
 51     for (i = 0; i < resizeIncrement; i++) {
 52         values.push(null);
 53     }
 54     this.capacity += resizeIncrement;
 55 };
 56 
 57 /**
 58  * Add a new value
 59  *
 60  * @param value
 61  *      the value
 62  */
 63 rock.js.Array.prototype.addValue = function (value) {
 64     var capacity = this.capacity;
 65     if (capacity == this.length) {
 66         this.resizeValues();
 67     }
 68 
 69     var values = this.values;
 70     values[this.length] = value;
 71     this.length++;
 72 };
 73 
 74 /**
 75  * Remove a value. This operation can be very expensive.
 76  *
 77  * @param value
 78  *      the value
 79  * @returns {Boolean} returns if the value existed in the array
 80  */
 81 rock.js.Array.prototype.removeValue = function (value) {
 82     var values = this.values;
 83     var length = this.length;
 84     var found = false;
 85     var i, pos;
 86     for (i = 0; i < length; i++) {
 87         if (values[i] === value) {
 88             found = true;
 89             pos = i;
 90             break;
 91         }
 92     }
 93 
 94     if (found) {
 95         // we maintain order
 96         for (i = pos; i < length - 1; i++) {
 97             values[i] = values[i + 1];
 98         }
 99         this.length--;
100     }
101 
102     return found;
103 };

105 /**
106  * Clear the array
107  *
108  * @param {Boolean} [setNulls]
109  *      set null on all the elements of the underlying array. Default will be false
110  */
111 rock.js.Array.prototype.clear = function (setNulls) {
112     this.length = 0;
113 
114     // This may be important because if you don't set nulls, there will be a reference
115     // to all existing elements. This will cause that these elements will not be
116     // released by garbage collector.
117     if (!rock.util.JsUtils.isNullOrUndefined(setNulls) && setNulls) {
118         var i;
119         var values = this.values;
120         for (i = 0; i < values.length; i++) {
121             values[i] = null;
122         }
123     }
124 };

126 /**
127  * Adjust the capacity to the current length
128  */
129 rock.js.Array.prototype.adjustCapacity = function () {
130     this.values.length = this.length;
131     this.capacity = this.length;
132 };
