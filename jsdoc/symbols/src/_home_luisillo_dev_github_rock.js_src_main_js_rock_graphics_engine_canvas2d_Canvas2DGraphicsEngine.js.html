 19 /**
 20  * @see rock.graphics.engine.IGraphicsEngine#clear
 21  * @override
 22  */
 23 rock.graphics.engine.canvas2d.Canvas2DGraphicsEngine.prototype.clear = function (color) {
 24     var canvas2D = this.context.getHTMLContext();
 25     canvas2D.beginPath();
 26     canvas2D.clearRect(0, 0, this.width, this.height);
 27     this.drawRectangle(0, 0, this.width, this.height, color);
 28 };

 30 /**
 31  * @see rock.graphics.engine.IGraphicsEngine#drawPoint
 32  * @override
 33  */
 34 rock.graphics.engine.canvas2d.Canvas2DGraphicsEngine.prototype.drawPoint = function (x, y, size, color) {
 35     var canvas2D = this.context.getHTMLContext();
 36     canvas2D.fillStyle = color.getAsRGBA();
 37     canvas2D.beginPath();
 38     canvas2D.arc(x, y, size/2, 0, 2 * Math.PI, true);
 39     canvas2D.fill();
 40 };

 42 /**
 43  * @see rock.graphics.engine.IGraphicsEngine#drawLine
 44  * @override
 45  */
 46 rock.graphics.engine.canvas2d.Canvas2DGraphicsEngine.prototype.drawLine = function (fromX, fromY, toX, toY, width, color) {
 47     var canvas2D = this.context.getHTMLContext();
 48 
 49     var adjustedFromX = this.adjustPixelPosition(fromX, width);
 50     var adjustedFromY = this.adjustPixelPosition(fromY, width);
 51     var adjustedToX = this.adjustPixelPosition(toX, width);
 52     var adjustedToY = this.adjustPixelPosition(toY, width);
 53 
 54     canvas2D.strokeStyle = color.getAsRGBA();
 55     canvas2D.lineWidth = width;
 56     canvas2D.beginPath();
 57     canvas2D.moveTo(adjustedFromX, adjustedFromY);
 58     canvas2D.lineTo(adjustedToX, adjustedToY);
 59     canvas2D.stroke();
 60 };

 62 /**
 63  * @see rock.graphics.engine.IGraphicsEngine#drawRectangle
 64  * @override
 65  */
 66 rock.graphics.engine.canvas2d.Canvas2DGraphicsEngine.prototype.drawRectangle = function
 67         (x, y, width, height, color, strokeWidth) {
 68     if (rock.util.JsUtils.isNullOrUndefined(strokeWidth) || strokeWidth < 1) {
 69         this.drawFilledRectangle(x, y, width, height, color);
 70     } else {
 71         this.drawStrokeRectangle(x, y, width, height, color, strokeWidth);
 72     }
 73 };

117 rock.graphics.engine.canvas2d.Canvas2DGraphicsEngine.prototype.drawStrokeRectangle = function
118         (x, y, width, height, color, strokeWidth) {
119     var canvas2D = this.context.getHTMLContext();
120     var adjustedX = this.adjustPixelPosition(x, strokeWidth);
121     var adjustedY = this.adjustPixelPosition(y, strokeWidth);
122 
123     canvas2D.strokeStyle = color.getAsRGBA();
124     canvas2D.lineWidth = strokeWidth;
125     canvas2D.beginPath();
126     canvas2D.rect(adjustedX, adjustedY, width, height);
127     canvas2D.stroke();
128 };

130 /**
131  * @see rock.graphics.engine.IGraphicsEngine#drawRoundedRectangle
132  * @override
133  */
134 rock.graphics.engine.canvas2d.Canvas2DGraphicsEngine.prototype.drawRoundedRectangle = function
135         (x, y, width, height, radius, color) {
136     var canvas2D = this.context.getHTMLContext();
137     var xmin = x;
138     var ymin = y;
139     var xmax = x + width;
140     var ymax = y + height;
141     canvas2D.fillStyle = color.getAsRGBA();
142     canvas2D.beginPath();
143     canvas2D.arc(xmin + radius, ymin + radius, radius, Math.PI, Math.PI * 1.5, false);
144     canvas2D.lineTo(xmax - radius, ymin);
145     canvas2D.arc(xmax - radius, ymin + radius, radius, Math.PI * 1.5, 2 * Math.PI, false);
146     canvas2D.lineTo(xmax, ymax - radius);
147     canvas2D.arc(xmax - radius, ymax - radius, radius, 0, Math.PI * 0.5, false);
148     canvas2D.lineTo(xmin + radius, ymax);
149     canvas2D.arc(xmin + radius, ymax - radius, radius, Math.PI * 0.5, Math.PI, false);
150     canvas2D.fill();
151 };

153 /**
154  * @see rock.graphics.engine.IGraphicsEngine#drawImage
155  * @override
156  */
157 rock.graphics.engine.canvas2d.Canvas2DGraphicsEngine.prototype.drawImage = function (image, x, y, width, height) {
158     var canvas2D = this.context.getHTMLContext();
159     canvas2D.beginPath();
160     if (rock.util.JsUtils.isNullOrUndefined(width) || rock.util.JsUtils.isNullOrUndefined(height)) {
161         canvas2D.drawImage(image.getHTMLImage(), x, y);
162     } else {
163         canvas2D.drawImage(image.getHTMLImage(), 0, 0, image.getWidth(), image.getHeight(), x, y,
164             width, height);
165     }
166 };

180 /**
181  * @see rock.graphics.engine.IGraphicsEngine#drawText
182  * @override
183  */
184 rock.graphics.engine.canvas2d.Canvas2DGraphicsEngine.prototype.drawText = function
185         (text, x, y, font, color, clippingWidth, clippingHeight) {
186     var canvas2D = this.context.getHTMLContext();
187     canvas2D.save();
188 
189     if (!rock.util.JsUtils.isNullOrUndefined(clippingWidth) && !rock.util.JsUtils.isNullOrUndefined(clippingHeight)) {
190         canvas2D.strokeStyle = 'rgba(255, 255, 255, 0)';
191         canvas2D.beginPath();
192         canvas2D.rect(x, y, clippingWidth, clippingHeight);
193         canvas2D.stroke();
194         canvas2D.clip();
195     }
196 
197     canvas2D.font = font.toCanvasFormat();
198     canvas2D.fillStyle = color.getAsRGBA();
199     canvas2D.beginPath();
200     canvas2D.fillText(text, x, y + font.getSize());
201     canvas2D.restore();
202 };
