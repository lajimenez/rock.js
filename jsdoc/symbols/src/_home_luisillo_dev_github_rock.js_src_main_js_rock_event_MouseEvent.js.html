  4  * Class for mouse events.

 10  * @extends rock.event.Event

 13 rock.event.MouseEvent = function (id) {
 14     rock.super_(this, [id]);
 15 
 16     this.x = null;
 17     this.y = null;
 18     this.DOMMouseEvent = null;
 19 };
 20 
 21 rock.extends_(rock.event.MouseEvent, rock.event.Event);
 22 
 23 /**
 24  * Return the mouse button that has done the action
 25  *
 26  * @returns the button
 27  */
 28 rock.event.MouseEvent.prototype.getButton = function () {
 29     var DOMMouseEvent = this.DOMMouseEvent;
 30     if (rock.util.JsUtils.isNullOrUndefined(DOMMouseEvent)) {
 31         return null;
 32     }
 33 
 34     var mouseButton = DOMMouseEvent.button;
 35     var button = null;
 36 
 37     if (mouseButton == rock.constants.HTML_DOM_EVENT_MOUSE_BUTTON_LEFT) {
 38         button = rock.constants.ROCK_EVENT_MOUSE_BUTTON_LEFT;
 39     } else if (mouseButton == rock.constants.HTML_DOM_EVENT_MOUSE_BUTTON_WHEEL) {
 40         button = rock.constants.ROCK_EVENT_MOUSE_BUTTON_WHEEL;
 41     } else if (mouseButton == rock.constants.HTML_DOM_EVENT_MOUSE_BUTTON_RIGHT) {
 42         button = rock.constants.ROCK_EVENT_MOUSE_BUTTON_RIGHT;
 43     }
 44 
 45     return button;
 46 };
