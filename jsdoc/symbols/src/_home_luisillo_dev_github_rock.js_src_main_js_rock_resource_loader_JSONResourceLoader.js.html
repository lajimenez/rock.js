1 rock.namespace('rock.resource.loader');
  2 
  3 /**
  4  * Class to load a JSON object
  5  *
  6  * @constructor
  7  * @extends rock.resource.loader.AbstractResourceLoader
  8  * @author Luis Alberto Jim√©nez
  9  */
 10 rock.resource.loader.JSONResourceLoader = function () {
 11     rock.super_(this, arguments);
 12 
 13     this.httpMethod = rock.constants.HTTP_METHOD_GET;
 14 };
 15 
 16 rock.extends_(rock.resource.loader.JSONResourceLoader,
 17     rock.resource.loader.AbstractResourceLoader);
 18 
 19 /**
 20  * @override
 21  * @see rock.resource.loader.AbstractResourceLoader#loadResource
 22  */
 23 rock.resource.loader.JSONResourceLoader.prototype.loadResource = function (elem) {
 24     var ajaxRequest = new rock.network.ajax.AjaxRequest(elem.getHTTPParams());
 25     ajaxRequest.addEventListener(
 26         rock.network.ajax.event.AjaxEvent.AJAX_EVENT, rock
 27             .createEventHandler(this, this.onAjaxEvent, [elem]));
 28     ajaxRequest.doCall();
 29 };
 30 
 31 /*
 32  * Ajax Event handler
 33  */
 34 rock.resource.loader.JSONResourceLoader.prototype.onAjaxEvent = function (ajaxEvent, elem) {
 35     if (!ajaxEvent.getError()) {
 36         var obj = JSON.parse(ajaxEvent.getResponseText());
 37         this.resourceLoaderManager.onResourceLoadEvent(true, elem, obj);
 38     } else {
 39         this.resourceLoaderManager.onResourceLoadEvent(false, elem, null);
 40     }
 41 };
