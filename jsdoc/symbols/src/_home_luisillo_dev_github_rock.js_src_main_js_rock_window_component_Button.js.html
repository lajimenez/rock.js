  3 /**
  4  * Button class
  5  *
  6  * @param {rock.window.Window} window
  7  *            the window where component exists
  8  * @param {String} text
  9  *            the text to show in label
 10  * @param {Boolean} [localize]
 11  *            if true, the text will be localized (so text should be the resource id of the localized string)
 12  * @param {rock.graphics.Font} [font]
 13  *            the font to use
 14  * @param {rock.graphics.Color} [color]
 15  *            the color to use
 16  *
 17  * @constructor
 18  * @extends rock.window.component.TextComponent
 19  *
 20  * @author Luis Alberto Jim√©nez
 21  */
 22 rock.window.component.Button = function (window, text, localize, font, color) {
 23     rock.super_(this, [window, text, localize, font, color]);
 24     this.width = rock.window.component.Component.AUTO_SIZE;
 25     this.height = rock.window.component.Component.AUTO_SIZE;
 26     this.MARGIN_WIDTH = 6;
 27     this.MARGIN_HEIGHT = 3;
 28     this.BORDER_WIDTH = 1;
 29     this.borderColor = this.application.getStyleManager().getColor(rock.window.style.StyleManager.COMPONENT_COLOR);
 30     this.onMouseOverColor = this.application.getStyleManager().getColor(rock.window.style.StyleManager.COMPONENT_COLOR_2);
 31     this.textPosition = null;
 32 
 33     this.addEventListener(rock.constants.ROCK_EVENT_MOUSE_OVER, rock.createEventHandler(this, this.handleOnMouseOver));
 34     this.addEventListener(rock.constants.ROCK_EVENT_MOUSE_LEAVE, rock.createEventHandler(this, this.handleOnMouseOut));
 35 };
 36 
 37 rock.extends_(rock.window.component.Button, rock.window.component.TextComponent);
 38 
 39 rock.window.component.Button.prototype.handleOnMouseOver = function (event) {
 40     this.redraw();
 41 };
 42 
 43 rock.window.component.Button.prototype.handleOnMouseOut = function (event) {
 44     this.redraw();
 45 };
 46 
 47 /**
 48  * @override
 49  * @see rock.window.component.Component#updateComponent
 50  */
 51 rock.window.component.Button.prototype.updateComponent = function () {
 52     this.localizeText();
 53     var text = this.localizedText;
 54     var font = this.font;
 55     var textWidth = rock.util.TextUtils.measureTextWidth(text, font);
 56     var textHeight = rock.util.TextUtils.measureTextHeight(text, font);
 57 
 58     if (this.width === rock.window.component.Component.AUTO_SIZE) {
 59         this.computedWidth = textWidth + 2 * this.MARGIN_WIDTH;
 60     } else {
 61         this.computedWidth = this.width;
 62     }
 63 
 64     if (this.height === rock.window.component.Component.AUTO_SIZE) {
 65         this.computedHeight = textHeight + 2 * this.MARGIN_HEIGHT;
 66     } else {
 67         this.computedHeight = this.height;
 68     }
 69 
 70     this.textPosition = rock.util.GeometryUtils.getPointMakesSameCenter(new rock.geometry.Point2(this.x, this.y),
 71         this.computedWidth, this.computedHeight, textWidth, textHeight);
 72 
 73 };
 74 
 75 rock.window.component.Button.prototype.getBackgroundColorToUse = function () {
 76     if (this.mouseOver) {
 77         return this.onMouseOverColor;
 78     } else {
 79         return this.backgroundColor;
 80     }
 81 };
 82 
 83 /**
 84  * @override
 85  * @see rock.graphics.engine.IDrawable#draw
 86  */
 87 rock.window.component.Button.prototype.draw = function (graphicsEngine) {
 88     this.updateComponent();
 89     graphicsEngine.drawRectangle(this.x, this.y, this.computedWidth, this.computedHeight,
 90         this.getBackgroundColorToUse());
 91     graphicsEngine.drawRectangle(this.x, this.y, this.computedWidth, this.computedHeight,
 92         this.borderColor, this.BORDER_WIDTH);
 93     var textPosition = this.textPosition;
 94     graphicsEngine.drawText(this.localizedText, textPosition.getX(), textPosition.getY(),
 95         this.font, this.color);
 96 };
