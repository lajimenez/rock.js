1 rock.namespace('rock.resource.loader');
  2 
  3 /**
  4  * Class to load an image resource
  5  *
  6  * @constructor
  7  * @extends rock.resource.loader.AbstractResourceLoader
  8  * @author Luis Alberto Jim√©nez
  9  */
 10 rock.resource.loader.ImageResourceLoader = function () {
 11     rock.super_(this, arguments);
 12 };
 13 
 14 rock.extends_(rock.resource.loader.ImageResourceLoader,
 15     rock.resource.loader.AbstractResourceLoader);
 16 
 17 /**
 18  * @override
 19  * @see rock.resource.loader.AbstractResourceLoader#loadResource
 20  */
 21 rock.resource.loader.ImageResourceLoader.prototype.loadResource = function (elem) {
 22     var HTMLImage = new Image();
 23     HTMLImage.id = elem.id;
 24 
 25     HTMLImage.onload = rock.createEventHandler(this, this.onLoadImage, [elem,
 26         HTMLImage]);
 27 
 28     HTMLImage.onerror = rock.createEventHandler(this, this.onErrorImage, [
 29         elem, HTMLImage]);
 30 
 31     HTMLImage.onabort = rock.createEventHandler(this, this.onAbortImage, [
 32         elem, HTMLImage]);
 33 
 34     HTMLImage.src = elem.getHTTPParams().getUrl();
 35 };
 36 
 37 /*
 38  * Event handler for an image
 39  */
 40 rock.resource.loader.ImageResourceLoader.prototype.onLoadImage = function (event, elem, HTMLImage) {
 41     this.callResourceLoader(event, elem, HTMLImage, true);
 42 };
 43 
 44 /*
 45  * Event handler for an image
 46  */
 47 rock.resource.loader.ImageResourceLoader.prototype.onErrorImage = function (event, elem, HTMLImage) {
 48     this.callResourceLoader(event, elem, HTMLImage, false);
 49 };
 50 
 51 /*
 52  * Event handler for an image
 53  */
 54 rock.resource.loader.ImageResourceLoader.prototype.onAbortImage = function (event, elem, HTMLImage) {
 55     this.callResourceLoader(event, elem, HTMLImage, false);
 56 };
 57 
 58 rock.resource.loader.ImageResourceLoader.prototype.callResourceLoader = function (event, elem, HTMLImage, success) {
 59     var image = new rock.graphics.Image(HTMLImage);
 60     this.resourceLoaderManager.onResourceLoadEvent(success, elem, image);
 61 };
