  3 /**
  4  * Base class for components.
  5  *
  6  * Rock framework components always call 'updateComponent' when drawing. That could be a bad idea if
  7  * this function do a lot of things, so you can only call this function when something has changed instead
  8  * calling it when drawing...
  9  *
 10  * STUDY: It will be nice to do some kind of study for improving components design. For example, TextComponent and
 11  * HTMLTextComponent are very similar... Probably there are some nice pattern to use... 'Decorator pattern' maybe?
 12  *
 13  * @param {rock.window.Window} window
 14  *      the window where component exists
 15  *
 16  * @constructor
 17  * @abstract
 18  * @extends rock.event.EventDispatcher
 19  * @augments rock.graphics.engine.IDrawable
 20  *
 21  * @author Luis Alberto Jim√©nez
 22  */
 23 rock.window.component.Component = function (window) {
 24     rock.super_(this);
 25 
 26     this.window = window;
 27     this.application = window.getApplication();
 28 
 29     // Id
 30     this.id = null;
 31 
 32     // Absolute position
 33     this.x = NaN;
 34     this.y = NaN;
 35 
 36     // Width and height
 37     this.width = NaN;
 38     this.height = NaN;
 39 
 40     this.computedWidth = NaN;
 41     this.computedHeight = NaN;
 42 
 43     // Appearance
 44     this.backgroundColor =
 45         this.application.getStyleManager().getColor(rock.window.style.StyleManager.COMPONENT_BACKGROUND_COLOR);
 46 
 47     this.mouseOver = false;
 48 
 49     this.application.addEventListener(rock.app.Application.LOCALE_CHANGED_EVENT,
 50         rock.createEventHandler(this, this.handleOnLocaleChanged));
 51 };

 86 /**
 87  * Check if the current coord hits the component
 88  *
 89  * @param x
 90  *            the coord x
 91  * @param y
 92  *            the coord y
 93  *
 94  * @returns {Boolean} if the coord hits the component
 95  */
 96 rock.window.component.Component.prototype.hit = function (x, y) {
 97     // We need to scale the mouse coord since all components will be scaled
 98     var scaleFactor = this.window.getApplication().getScaleFactor();
 99     var scaledX = x / scaleFactor;
100     var scaledY = y / scaleFactor;
101 
102     return this.x <= scaledX && scaledX <= (this.x + this.computedWidth)
103         && this.y <= scaledY && scaledY <= (this.y + this.computedHeight);
104 };

106 /**
107  * Handle a system mouse event
108  *
109  * @param eventId
110  *            the event ID
111  * @param x
112  *            the x click
113  * @param y
114  *            the y click
115  * @param mouseEvent
116  *            the event thrown by the DOM
117  */
118 rock.window.component.Component.prototype.handleMouseSystemEvent = function (eventId, x, y, mouseEvent) {
119     var hit = this.hit(x, y);
120 
121     if (hit) {
122         this.dispatchEvent(rock.window.WindowSystem.createMouseEvent(eventId, x, y, mouseEvent));
123     }
124 
125     if (eventId === rock.constants.ROCK_EVENT_MOUSE_MOVE) {
126         this.handleMouseOutMouseOverEvent(hit, x, y, mouseEvent);
127     }
128 };
129 
130 /**
131  * Handle mouse over and mouse out events
132  *
133  * @param hit
134  *            if the mouse is over the component
135  * @param x
136  *            the x click
137  * @param y
138  *            the y click
139  * @param mouseEvent
140  *            the event thrown by the DOM
141  */
142 rock.window.component.Component.prototype.handleMouseOutMouseOverEvent = function (hit, x, y, mouseEvent) {
143     if (hit && !this.mouseOver) {
144         this.mouseOver = true;
145         this.dispatchEvent(rock.window.WindowSystem.createMouseEvent(rock.constants.ROCK_EVENT_MOUSE_OVER, x, y, mouseEvent));
146     }
147     else if (!hit && this.mouseOver) {
148         this.mouseOver = false;
149         this.dispatchEvent(rock.window.WindowSystem.createMouseEvent(rock.constants.ROCK_EVENT_MOUSE_LEAVE, x, y, mouseEvent));
150     }
151 };

177 /**
178  * Redraw the component
179  * @protected
180  */
181 rock.window.component.Component.prototype.redraw = function () {
182     var window = this.window;
183     var windowSystem = window.getWindowSystem();
184     // If the component we want to redraw is not in the current window,
185     // we don't have to redraw it
186     if (window != windowSystem.getCurrentWindow()) {
187         return;
188     }
189 
190     this.draw(windowSystem.getGraphicsEngine());
191 };

255 /**
256  * Set the width of the component (use 'rock.window.component.Component.AUTO_SIZE' for auto sizing)
257  * If any subcomponent doesn't allow auto size, it must override this function and throw an
258  * invalid component operation.
259  *
260  * @param width the value
261  *
262  * ATTENTION: Setting width may or not compute the real width
263  * (you must assure that 'updateComponent' has been called if you want get real width)
264  */
265 rock.window.component.Component.prototype.setWidth = function(width) {
266     if (width < rock.window.component.Component.AUTO_SIZE) {
267         throw new rock.error.RockError(rock.resource.rockResourceManager.getString('INVALID_COMPONENT_OPERATION'));
268     }
269     this.width = width;
270 };
