  3 /**
  4  * Window to show loading process
  5  *
  6  * @param  {rock.app.Application} application
  7  *            the application where this window will be part
  8  *
  9  * @constructor
 10  * @extends rock.window.Window
 11  * @author Luis Alberto Jim√©nez
 12  */
 13 rock.window.ResourceLoaderWindow = function (application) {
 14     rock.super_(this, arguments);
 15     this.resourceLoaderManager = new rock.resource.loader.ResourceLoaderManager();
 16     this.resourceLoaderManager.addEventListener(
 17         rock.resource.event.ResourceLoadedEvent.RESOURCE_LOADED,
 18         rock.createEventHandler(this, this.resourceLoaded));
 19 
 20     this.resourceLoaderManager
 21         .addEventListener(
 22         rock.resource.event.ResourceLoaderComplete.RESOURCE_LOADER_COMPLETE,
 23         rock.createEventHandler(this,
 24             this.resourceLoaderManagerComplete));
 25 
 26     // Space between Label and ProgressBar
 27     this.margin = 5;
 28 
 29     this.progressBar = new rock.window.component.ProgressBarComponent(this);
 30     this.setProgressBarWidth(this.windowSystem.getWidth() / 2);
 31     this.setProgressBarHeight(30);
 32 
 33     this.addComponent(this.progressBar);
 34 
 35     this.label = new rock.window.component.Label(this, rock.resource.rockResourceManager.getString('RESOURCE_WINDOW_LOADING_LABEL'));
 36     this.addComponent(this.label);
 37 };
 38

 71 rock.window.ResourceLoaderWindow.prototype.resourceLoaded = function (event) {

 82 /**
 83  * Compute position of progress bar to set at center
 84  */
 85 rock.window.ResourceLoaderWindow.prototype.updateComponents = function () {
 86     var zeroPoint = new rock.geometry.Point2(0, 0);
 87     var drawPoint = rock.util.GeometryUtils.getPointMakesSameCenter(zeroPoint,
 88         this.windowSystem.getWidth(), this.windowSystem.getHeight(),
 89         this.progressBar.getComputedWidth(), this.progressBar.getComputedHeight());
 90 
 91     this.progressBar.setX(drawPoint.getX());
 92     this.progressBar.setY(drawPoint.getY());
 93 
 94     var label = this.label;
 95     var textWidth = label.getComputedWidth();
 96     var textHeight = label.getComputedHeight();
 97 
 98     drawPoint = rock.util.GeometryUtils.getPointMakesSameCenter(zeroPoint,
 99         this.windowSystem.getWidth(), this.windowSystem.getHeight(),
100         textWidth, textHeight);
101 
102     this.label.setX(drawPoint.getX());
103     this.label.setY(this.progressBar.getY() - (textHeight + this.margin));
104 };
