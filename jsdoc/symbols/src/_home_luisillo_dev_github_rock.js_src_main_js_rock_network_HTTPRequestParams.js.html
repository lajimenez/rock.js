  3 /**
  4  * Parameters for using when doing an HTTP request
  5  *
  6  * @param url
  7  *            the url to do the call
  8  * @param method
  9  *            HTTP method {GET, POST}
 10  * @param {object} HTTPParams
 11  *            HTTP parameters. Object containing key pair values
 12  *
 13  * @constructor
 14  * @author Luis Alberto Jim√©nez
 15  */
 16 rock.network.HTTPRequestParams = function (url, method, HTTPParams) {
 17     this.url = url;
 18     this.method = method;
 19     if (rock.util.JsUtils.isNullOrUndefined(this.method)){
 20         this.method = rock.constants.HTTP_METHOD_GET;
 21     }
 22     this.HTTPParams = HTTPParams;
 23 };
