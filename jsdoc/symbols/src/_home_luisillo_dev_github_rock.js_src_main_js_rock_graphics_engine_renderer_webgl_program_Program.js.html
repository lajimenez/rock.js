  3 /**
  4  * Represents a WebGL program
  5  *
  6  * @param {rock.graphics.engine.webgl.WebGLContext} glContext
  7  *         the WebGL context
  8  * @param {String} vertexShaderSrc
  9  *         source for the vertex shader
 10  * @param {String} fragmentShaderSrc
 11  *         source for the fragment shader
 12  *
 13  * @constructor
 14  *
 15  * @author Luis Alberto Jim√©nez
 16  */
 17 rock.graphics.engine.renderer.webgl.program.Program = function (glContext, vertexShaderSrc, fragmentShaderSrc) {
 18     this.glContext = glContext;
 19     this.vertexShader = null;
 20     this.fragmentShader = null;
 21     this.linked = false;
 22     this.glProgram = null;
 23     this.createProgram(vertexShaderSrc, fragmentShaderSrc);
 24 
 25     // Handle context events
 26     var app = glContext.getApplication();
 27     app.addEventListener(rock.constants.ROCK_EVENT_CONTEXT_RESTORED, rock.createEventHandler(this, this.onContextRestored));
 28 };
 29 
 30 /**
 31  * Set the renderer as active
 32  */
 33 rock.graphics.engine.renderer.webgl.program.Program.prototype.useProgram = function () {
 34     var gl = this.glContext.getHTMLContext();
 35     gl.useProgram(this.getGlProgram());
 36 };
 37 
 38 /**
 39  * Create the WebGL program. Shaders will be created and compiled and the program will be linked.
 40  *
 41  * @param vertexShaderSrc
 42  * @param fragmentShaderSrc
 43  *
 44  * @private
 45  */
 46 rock.graphics.engine.renderer.webgl.program.Program.prototype.createProgram = function (vertexShaderSrc,
 47                                                                                         fragmentShaderSrc) {
 48     this.vertexShader = null;
 49     this.fragmentShader = null;
 50     this.linked = false;
 51     this.glProgram = null;
 52 
 53     this.createShaders(vertexShaderSrc, fragmentShaderSrc);
 54     this.createProgramAndLink();
 55 };
 56 
 57 /**
 58  * Create the shaders
 59  *
 60  * @private
 61  */
 62 rock.graphics.engine.renderer.webgl.program.Program.prototype.createShaders =
 63         function (vertexShaderSrc, fragmentShaderSrc) {
 64     var glContext = this.glContext;
 65     this.vertexShader =
 66         new rock.graphics.engine.renderer.webgl.program.Shader(
 67             glContext, vertexShaderSrc, rock.graphics.engine.renderer.webgl.program.Shader.VERTEX_SHADER);
 68     this.fragmentShader =
 69         new rock.graphics.engine.renderer.webgl.program.Shader(
 70             glContext, fragmentShaderSrc, rock.graphics.engine.renderer.webgl.program.Shader.FRAGMENT_SHADER);
 71 };
 72 
 73 /**
 74  * Link the program
 75  *
 76  * @private
 77  */
 78 rock.graphics.engine.renderer.webgl.program.Program.prototype.createProgramAndLink = function () {
 79     var gl = this.glContext.getHTMLContext();
 80 
 81     var vertexShader = this.vertexShader;
 82     var fragmentShader = this.fragmentShader;
 83     if (!vertexShader.isCompiled()) {
 84         vertexShader.compile();
 85     }
 86     if (!fragmentShader.isCompiled()) {
 87         fragmentShader.compile();
 88     }
 89 
 90     this.glProgram = rock.util.WebGlUtils.createProgram(gl, vertexShader.getGlShader(), fragmentShader.getGlShader());
 91     this.linked = true;
 92 };
 93 
 94 /**
 95  * Function for handling context restored
 96  *
 97  * @private
 98  */
 99 rock.graphics.engine.renderer.webgl.program.Program.prototype.onContextRestored = function (event) {
100     this.restore();
101 };
102 
103 /**
104  * Restore the program. The WebGL program will be re-created
105  */
106 rock.graphics.engine.renderer.webgl.program.Program.prototype.restore = function () {
107     var vertexShader = this.vertexShader;
108     var fragmentShader = this.fragmentShader;
109 
110     if (vertexShader == null || fragmentShader == null) {
111         return;
112     }
113 
114     this.createProgram(vertexShader.getSrc(), fragmentShader.getSrc());
115 };
