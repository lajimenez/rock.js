 10 rock.test.TestRunner = function () {
 11 };
 12 
 13 /**
 14  * Test functions mush start with this prefix
 15  *
 16  * @constant
 17  */
 18 rock.test.TestRunner.TEST_PREFIX = 'test';
 19 
 20 /**
 21  * Run all tests in a test suite
 22  *
 23  * @param {rock.test.TestSuite} testSuite
 24  *      test to be executed
 25  * @returns {rock.test.TestSuiteResult} information about the test execution
 26  */
 27 rock.test.TestRunner.prototype.runTestSuite = function (testSuite) {
 28     if (!rock.instanceof_(testSuite, rock.test.TestSuite)) {
 29         throw new rock.error.RockError(rock.resource.rockResourceManager.getString('TEST_INVALID_TYPE'));
 30     }
 31 
 32     var testSuiteResult = new rock.test.TestSuiteResult();
 33 
 34     var propertyName, property;
 35     for (propertyName in testSuite) {
 36         property = testSuite[propertyName];
 37         if (rock.isFunction(property)
 38             && rock.util.JsUtils.stringStartsWith(propertyName, rock.test.TestRunner.TEST_PREFIX)) {
 39             testSuiteResult.addResultForTest(propertyName, this.runTest(testSuite, propertyName));
 40         }
 41     }
 42 
 43     return testSuiteResult;
 44 };
 45 
 46 /*
 47  * Run test
 48  *
 49  * @param testSuite
 50  *      test suite object
 51  * @param testName
 52  *      function in test suite to be executed
 53  *
 54  * @returns {boolean} if the test has failed
 55  */
 56 rock.test.TestRunner.prototype.runTest = function (testSuite, testName) {
 57     testSuite.reset();
 58     testSuite.setUp();
 59     try {
 60         testSuite[testName]();
 61     } catch (error) {
 62         testSuite.markCurrentTestAsFailed();
 63     }
 64     testSuite.tearDown();
 65 
 66     return testSuite.hasCurrentTestFailed();
 67 };
