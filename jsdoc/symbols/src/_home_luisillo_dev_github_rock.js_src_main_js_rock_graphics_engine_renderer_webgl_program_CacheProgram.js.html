  3 /**
  4  * Represents a WebGL program. This class cache some information so we can take some advantage when using it.
  5  *
  6  * @param {rock.graphics.engine.webgl.WebGLContext} glContext
  7  *         the WebGL context
  8  * @param {String} vertexShaderSrc
  9  *         source for the vertex shader
 10  * @param {String} fragmentShaderSrc
 11  *         source for the fragment shader
 12  * @param {Array} attributes
 13  *          list of attributes in program
 14  * @param {Array} uniforms
 15  *          list of uniform in program
 16  * @param {Array} textures
 17  *          list of textures in program
 18  * @param {Boolean} [provisioningBuffers]
 19  *          if true, caching will also create buffers
 20  * @param {Boolean} [provisioningTextures]
 21  *          if true, caching will also create textures
 22  *
 23  * @constructor
 24  * @extends rock.graphics.engine.renderer.webgl.program.Program
 25  *
 26  * @author Luis Alberto Jim√©nez
 27  */
 28 rock.graphics.engine.renderer.webgl.program.CacheProgram = function (glContext, vertexShaderSrc, fragmentShaderSrc,
 29                                                                 attributes, uniforms, textures,
 30                                                                 provisioningBuffers, provisioningTextures) {
 31     rock.super_(this, [glContext, vertexShaderSrc, fragmentShaderSrc]);
 32 
 33     this.attributes = null;
 34     this.uniforms = null;
 35     this.textures = null;
 36 
 37     this.indexes = null;
 38 
 39     var createBuffers = provisioningBuffers;
 40     if (rock.util.JsUtils.isNullOrUndefined(createBuffers)) {
 41         createBuffers = true;
 42     }
 43     var createTextures = provisioningTextures;
 44     if (rock.util.JsUtils.isNullOrUndefined(createTextures)) {
 45         createTextures = true;
 46     }
 47 
 48     this.cache(attributes, uniforms, textures, createBuffers, createTextures);
 49 
 50     // Used for restore
 51     this.createBuffers = createBuffers;
 52     this.createTextures = createTextures;
 53 };

 69 rock.graphics.engine.renderer.webgl.program.CacheProgram.prototype.cache = function (attributes, uniforms, textures,
 70                                                                                      createBuffers, createTextures) {

 77 /**
 78  * Cache the location and create a buffer for all attributes
 79  *
 80  * @param {Array} attributes
 81  *          list of attributes in program
 82  * @param createBuffers
 83  *
 84  * @private
 85  */
 86 rock.graphics.engine.renderer.webgl.program.CacheProgram.prototype.cacheAttributes = function (attributes, createBuffers) {
 87     var gl = this.glContext.getHTMLContext();
 88     var glProgram = this.glProgram;
 89     var buffer = null;
 90     var i, name, location;
 91 
 92     this.attributes = [];
 93 
 94     for ( i = 0; i < attributes.length; i++) {
 95         name = attributes[i];
 96         location =  gl.getAttribLocation(glProgram, name);
 97         if (createBuffers) {
 98             buffer = gl.createBuffer();
 99         }
100         this.attributes.push({
101             name: name,
102             location: location,
103             buffer: buffer
104         });
105     }
106 };

214 rock.graphics.engine.renderer.webgl.program.CacheProgram.prototype.getByName = function (name, array) {
215     var i, object;

217         object = array[i];

219             return object;
220         }
221     }

237 /**
238  * Populate an attribute
239  *
240  * @param {String} name
241  *      attribute name to populate
242  * @param {Float32Array} values
243  *      values used to populate attribute
244  * @param {Number} dimensions
245  *      dimensions of the values
246  */
247 rock.graphics.engine.renderer.webgl.program.CacheProgram.prototype.populateAttribute = function (name, values, dimensions) {
248     var gl = this.glContext.getHTMLContext();
249     var attribute = this.getByName(name, this.attributes);
250     gl.bindBuffer(gl.ARRAY_BUFFER, attribute.buffer);
251     gl.bufferData(gl.ARRAY_BUFFER, values, gl.STATIC_DRAW);
252     gl.vertexAttribPointer(attribute.location, dimensions, gl.FLOAT, gl.FALSE, 0, 0);
253 };

267 /**
268  * Populate a texture with an image
269  *
270  * @param {String} name
271  *      texture name to populate
272  * @param {rock.graphics.Image} image
273  *      image used to populate texture
274  */
275 rock.graphics.engine.renderer.webgl.program.CacheProgram.prototype.populateTexture = function (name, image) {
276     var gl = this.glContext.getHTMLContext();
277     var texture = this.getByName(name, this.textures);
278 
279     gl.activeTexture(gl.TEXTURE0 + texture.textureUnit);
280     gl.bindTexture(gl.TEXTURE_2D, texture.texture);
281     gl.texParameteri(gl.TEXTURE_2D, gl.TEXTURE_WRAP_S, gl.CLAMP_TO_EDGE);
282     gl.texParameteri(gl.TEXTURE_2D, gl.TEXTURE_WRAP_T, gl.CLAMP_TO_EDGE);
283     gl.texParameteri(gl.TEXTURE_2D, gl.TEXTURE_MIN_FILTER, gl.NEAREST);
284     gl.texParameteri(gl.TEXTURE_2D, gl.TEXTURE_MAG_FILTER, gl.NEAREST);
285     gl.texImage2D(gl.TEXTURE_2D, 0, gl.RGBA, gl.RGBA, gl.UNSIGNED_BYTE, image.getHTMLImage());
286 
287     gl.uniform1i(texture.location, texture.textureUnit);
288 };

318 /**
319  * Configure a texture using an also generated glTexture
320  *
321  * @param {String} name
322  *      texture name to populate
323  * @param {WebGLTexture} glTexture
324  *      the texture
325  */
326 rock.graphics.engine.renderer.webgl.program.CacheProgram.prototype.updateTexture = function (name, glTexture) {
327     var gl = this.glContext.getHTMLContext();
328     var texture = this.getByName(name, this.textures);
329     gl.activeTexture(gl.TEXTURE0 + texture.textureUnit);
330     gl.bindTexture(gl.TEXTURE_2D, glTexture);
331     gl.uniform1i(texture.location, texture.textureUnit);
332 };
