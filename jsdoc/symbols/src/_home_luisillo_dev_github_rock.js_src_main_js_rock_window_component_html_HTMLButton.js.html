  3 /**
  4  * Button class
  5  *
  6  * @param {rock.window.Window} window
  7  *            the window where component exists
  8  * @param text
  9  *            the text to show in button

 12  * @param {rock.graphics.Font} [font]
 13  *            the font to use
 14  * @param {rock.graphics.Color} [color]
 15  *            the color to use
 16  *
 17  * @constructor
 18  * @extends rock.window.component.html.HTMLTextComponent
 19  *
 20  * @author Luis Alberto Jim√©nez
 21  */
 22 rock.window.component.html.HTMLButton = function (window, text, localize, font, color) {
 23     rock.super_(this, [window, text, localize, font, color, rock.constants.INPUT_TAG]);
 24     this.HTMLElement.type = rock.constants.INPUT_TYPE_BUTTON;
 25 };
 26 
 27 rock.extends_(rock.window.component.html.HTMLButton, rock.window.component.html.HTMLTextComponent);
 28 
 29 /**
 30  * @override
 31  * @see rock.window.component.Component#updateComponent
 32  */
 33 rock.window.component.html.HTMLButton.prototype.updateComponent = function () {
 34     var style = this.style;
 35     var scaleFactor = this.window.getApplication().getScaleFactor();
 36     rock.window.component.html.HTMLComponent.updateHTMLStyleSize(style, this.width, this.height, scaleFactor);
 37     rock.window.component.html.HTMLComponent.updateHTMLStyleFont(style, this.font, this.color, scaleFactor);
 38     rock.window.component.html.HTMLComponent.updateHTMLBackground(style, this.backgroundColor);
 39 
 40     this.localizeText();
 41     this.HTMLElement.value = this.localizedText;
 42 
 43     this.computeSize();
 44 };
