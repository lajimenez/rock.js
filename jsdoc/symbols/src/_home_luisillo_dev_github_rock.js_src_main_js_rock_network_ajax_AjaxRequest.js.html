  3 /**
  4  * Class to use when doing an ajax call
  5  *
  6  * @param {rock.network.HTTPRequestParams} params
  7  *            parameters for the call
  8  * @constructor
  9  * @author Luis Alberto Jim√©nez
 10  */
 11 rock.network.ajax.AjaxRequest = function (params) {
 12     rock.super_(this);
 13 
 14     this.url = params.getUrl();
 15     this.method = params.getMethod();
 16     this.httpParams = params.getHTTPParams();
 17 
 18     this.httpRequest = null;
 19 
 20 };
 21 
 22 rock.extends_(rock.network.ajax.AjaxRequest, rock.event.EventDispatcher);
 23 
 24 /**
 25  * Create an http request
 26  */
 27 rock.network.ajax.AjaxRequest.prototype.createHttpRequest = function () {
 28     var request = null;
 29 
 30     if (window.XMLHttpRequest && window.FormData) {
 31         request = new XMLHttpRequest();
 32     } else {
 33         throw new rock.error.RockError(rock.resource.rockResourceManager.getString('AJAX_NOT_SUPPORTED'));
 34     }
 35 
 36     return request;
 37 };
 38 
 39 /**
 40  * Do an ajax call
 41  */
 42 rock.network.ajax.AjaxRequest.prototype.doCall = function () {
 43     this.httpRequest = this.createHttpRequest();
 44     var httpRequest = this.httpRequest;
 45     httpRequest.onreadystatechange = rock.createEventHandler(this,
 46         this.processRequest, []);
 47 
 48     var params = this.getParamsURLFormat(this.httpParams);
 49     var url = this.url;
 50 
 51     if (!rock.isNullOrUndefined(params) && this.method == rock.constants.HTTP_METHOD_GET) {
 52         url = this.concatParamToURL(url, params);
 53     }
 54 
 55     httpRequest.open(this.method, url, true);
 56 
 57     var data = null;
 58     if (!rock.isNullOrUndefined(params) && this.method == rock.constants.HTTP_METHOD_POST) {
 59         data = params;
 60         httpRequest.setRequestHeader("Content-type", "application/x-www-form-urlencoded");
 61     }
 62 
 63     httpRequest.send(data);
 64 };
 65 
 66 rock.network.ajax.AjaxRequest.prototype.concatParamToURL = function (url, params) {
 67     var urlWithParams = url;
 68     if (!rock.util.JsUtils.stringContains(url, '?')) {
 69         urlWithParams = urlWithParams + '?' + params;
 70     } else if (rock.util.JsUtils.stringEndsWith(url, '&') || rock.util.JsUtils.stringEndsWith(url, '?')) {
 71         urlWithParams = urlWithParams + params;
 72     } else {
 73         urlWithParams = urlWithParams + '&' + params;
 74     }
 75     return urlWithParams;
 76 };
 77 
 78 rock.network.ajax.AjaxRequest.prototype.getParamsURLFormat = function (params) {
 79     if (rock.util.JsUtils.isNullOrUndefined(params)) {
 80         return null;
 81     }
 82 
 83     var urlParams = '';
 84     var property;
 85     var first = true;
 86     var atLeastOne = false;
 87     for (property in params) {
 88         if (params.hasOwnProperty(property)) {
 89             if (first) {
 90                 first = false;
 91             } else {
 92                 urlParams = urlParams + '&';
 93             }
 94             urlParams = urlParams + property + '=' + encodeURI(params[property]);
 95             atLeastOne = true;
 96         }
 97     }
 98 
 99     if (atLeastOne) {
100         return urlParams;
101     } else {
102         return null;
103     }
104 };

106 /**
107  * Cancel the request
108  */
109 rock.network.ajax.AjaxRequest.prototype.cancelRequest = function () {
110     this.httpRequest.abort();
111 };
112 
113 /**
114  * Process the request
115  */
116 rock.network.ajax.AjaxRequest.prototype.processRequest = function () {
117     if (this.httpRequest.readyState == 4) {
118         var responseText = null;
119         var error = true;
120         if (this.httpRequest.status == 200) {
121             // responseXML ??
122             responseText = this.httpRequest.responseText;
123             error = false;
124         }
125         var ajaxEvent = new rock.network.ajax.event.AjaxEvent(responseText,
126             error, this.httpRequest);
127         this.dispatchEvent(ajaxEvent);
128     }
129 };
